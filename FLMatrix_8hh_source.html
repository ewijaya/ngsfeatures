12 * Purpose: Provide a fixed matrix (2D array) class that checks for the validity of indices.
13 *
14 * Description: FLMatrix is a fixed length 2D array that can also be efficiently accessed as a 1D array.
15 * The operator () is used to access as a 2D array.
16 * The operator [] is used to access as a 1D array.
17 * e.g. FLMatrix<double> fml( 2, 3 );
18 * fml( 1, 2 ) = 3.2;
19 * fml[ 2*1+2 ] = 3.2; // these two lines are equivalent.
20 *
21 * Another way to access elements is to get a column pointer with getColPtr( size_t index0 ).
22 *
23 *
24 * FLMatrixFast is a derived class that avoids a small overhead by not checking
25 * the index passed to operator [], operator (), and getColPtr for being too large.
26 * However, when debugging FLARRAY_ALWAYS_CHECK_INDICES can be defined to force index
27 checking with FLMatrixFast too.
28 *
29 * FLNumMatrix is an extension of FLMatrix which assumes T can do arithmetic. It supplies
30 * methods such as sum() which returns the summation of the elements.
31 *
32 * FLNumMatrixFast is exactly analogous to FLMatrixFast. In fact I tried to have it multiple
33 * inherit from FLNumMatrix and FLMatrixFast -- but I couldn't get that to work...
34 *
35 */

175 FLMatrix<T> submatrix( const size_t startRow, const size_t endRow, const size_t startCol, const size_t endCol ) const{
176 // Return a copy of the submatrix from startRow to endRow and startCol to endCol (inclusive).
177 assert( startRow <= endRow );
178 assert( startCol <= endCol );
179 assert( validIndex( startRow, startCol ) );
180 assert( validIndex( endRow, endCol ) );
181 FLMatrix<T> retVal( endRow - startRow + 1, endCol - startCol + 1 );
182 for( size_t i = startRow; i <= endRow; ++i ){
183 for( size_t j = startCol; j <= endCol; ++j ){
184 retVal( i-startRow, j-startCol ) = this->operator()( i, j );
185 }
186 }
187 return retVal;
188 }

293 for( size_t i = 0; i < FLMatrix<T>::s; ++i ) FLMatrix<T>::m[i] += el;
294 return( *this );

296 FLNumMatrix& operator+=( const FLNumMatrix& flm ){
297 assert( FLMatrix<T>::s0 == flm.s0 );
298 assert( FLMatrix<T>::s1 == flm.s1 );
299 for( size_t i = 0; i < FLMatrix<T>::s; ++i ) FLMatrix<T>::m[i] += flm.m[i];
300 return( *this );
301 }
302 T sum(){
303 T ans = 0;
304 for( size_t i = 0; i < FLMatrix<T>::s; ++i ) ans += FLMatrix<T>::m[i];
305 return ans;
306 }
307
308 // dimension is really enum {0, 1}, choosing to sum over the current
309 // row or column.
310 T sum( unsigned char dimension, unsigned int index ){
311 assert( dimension < 2 );
312 T ans = 0;
313 if( dimension==0 ){
314 unsigned int base = index;
315 for( size_t i = 0; i < FLMatrix<T>::size0(); ++i ){
316 ans += FLMatrix<T>::m[ base ];
317 base += FLMatrix<T>::s1;
318 }
319 return ans;
320 }else{
321 unsigned int base = index * FLMatrix<T>::s1;
322 for( size_t i = 0; i < FLMatrix<T>::size1(); ++i ){
323 ans += FLMatrix<T>::m[ base + i ];
324 }
325 }
326 return ans;
327 }
328
329 // dimension is really enum {0, 1}, choosing to divide from elements
330 // of the current row or column.
331 T divide( unsigned char dimension, unsigned int index, T divisor ){
332 assert( dimension < 2 );
333 T ans = 0;
334 if( dimension==0 ){
335 unsigned int base = index;
336 for( size_t i = 0; i < FLMatrix<T>::size0(); ++i ){
337 FLMatrix<T>::m[ base ] /= divisor;
338 base += FLMatrix<T>::s1;
339 }
340 return ans;
341 }else{
342 unsigned int base = index * FLMatrix<T>::s1;
343 for( size_t i = 0; i < FLMatrix<T>::size1(); ++i ){
344 FLMatrix<T>::m[ base + i ] /= divisor;
345 }
346 }
347 return ans;
348 }
349 void normalize( unsigned char dimension ){ // normalize row or column
350 assert( dimension < 2 );
351 size_t sumSize;
352 if( dimension == 0 ) sumSize = FLMatrix<T>::size1();
353 else sumSize = FLMatrix<T>::size0();
354 for( unsigned int i = 0; i < sumSize; ++i ){
355 divide( dimension, i, sum( dimension, i ) );
356 }
357 }
