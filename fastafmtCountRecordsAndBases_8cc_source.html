23baseAndRecordCountPairT
24fastafmtCountRecordsAndBases( std::istream& fastaInputStream,
25 const char commentChar /* no effect when set to
26 FASTAFMT_RECORD_MARKER_CHAR */
27 ){
28 std::string line = "";
29 unsigned int numRecords = 0;
30 unsigned int numBases = 0;
31 while( std::getline( fastaInputStream, line ) ){
32 if( line.empty() ) continue; /* not actually necessary for usual the
33 implementation of std::string */
34 if( line[0] == FASTAFMT_RECORD_MARKER_CHAR ){
35 ++numRecords;
36 }else if( line[0] == commentChar ){
37 continue;
38 }else{ // body line. We treat a-zA-Z as bases.
39 for( size_t i = 0; i < line.size(); ++i ){
40 char c( line[0] );
41 if( ((c >= 'a') && (c <= 'z')) ||
42 ((c >= 'A') && (c <= 'Z')) ){
43 ++numBases;
44 }
45 }
46 }
47 } // get next line.
48
49 return std::make_pair( numRecords, numBases );
50}
