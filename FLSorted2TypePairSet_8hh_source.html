79 /* **************************************************
80 * Construct from text file. Format of text file is
81 * key1 value [value]...
82 * key2 value [value]...
83 *
84 * The fields should be tab separated.
85 * Each key should appear as a key in only one line
86 * ************************************************** */
87 void readFromTextFile( std::ifstream& ifs, const std::string& delimiter="\t" );
88
89 bool operator() ( const T0& lhs, const T1& rhs ) const{
90 if( !lhsToIndex.has(lhs) ) return false;
91 size_t rhIndex = lhsToIndex( lhs );
92 return lhsIndexToRhsSorted[rhIndex].has( rhs );
93 }
94
95
96 // data accessed by operator<<, would prefer to declare as private if possible.
97 FLSorted<T0> lhsToIndex;
98 FLArray< FLSorted<T1> > lhsIndexToRhsSorted;
99private:
100 void _initLhsIndexToRhsSorted( const std::vector<T0>& initialLhsList,
101 const std::vector< std::vector<T1> >& correspondingRightHandSideList ){
102
103 lhsIndexToRhsSorted.setSize( lhsToIndex.size() );
104 for( size_t i = 0; i < lhsToIndex.size(); ++i ){
105
106 // the indices of initialLhsList and correspondingRightHandSideList should be in sync.
107 size_t indexAfterSorting = lhsToIndex( initialLhsList[i] );
108
109 lhsIndexToRhsSorted[indexAfterSorting].assign( correspondingRightHandSideList[i].begin(),
110 correspondingRightHandSideList[i].end() );
111 }
112 }
113}; // end class FLSorted2TypePairSet

116template <typename T0, typename T1>
117inline void FLSorted2TypePairSet<T0, T1>::readFromTextFile( std::ifstream& ifs,
118 const std::string& delimiter ){
119 if( lhsToIndex.size() ){
120 std::cout << __FILE__ << ":" << __LINE__
121 << " Programming Error: FLSorted2TypePairSet, expected pair set to be empty when calling readFromTextFile\n";
122 exit( -1 );
123 }
124
125 std::string line;
126 {
127 std::vector<T0> lhsVector;
128 while( (line = perlish::slurpLine( ifs )).size() ){
129 lhsVector.push_back( perlish::split( delimiter, line )[0] );
130 }
131
132 lhsToIndex = lhsVector;
133
134 GDB_ASSERTF( lhsToIndex.size() == lhsVector.size(),
135 "lhsToIndex.size() = %zu, â‰  lhsVector.size() = %zu\n"
136 "Perhaps an lhs term (first field in line) was duplicated?",
137 lhsToIndex.size(), lhsVector.size() );
138
139 }
140 lhsIndexToRhsSorted.setSize( lhsToIndex.size() );
141 ifs.clear(); ifs.seekg( 0, std::ios::beg );
142 while( (line = perlish::slurpLine( ifs )).size() ){
143 std::vector<std::string> fields = perlish::split( delimiter, line );
144 T0 key = fields[0];
145
146 assert( lhsToIndex.has( key ) );
147 size_t rhIndex = lhsToIndex( key );
148 lhsIndexToRhsSorted[rhIndex].assign( fields.begin()+1, fields.end() );
149 }
150} // end readFromTextFile

179=head1 NAME
180
181FLSorted2TypePairSet
182
183=head1 SYNOPSIS
184
185Template class, implementing a set of ordered pairs.
186
187=head1 DESCRIPTION
188
189B<FLSorted2TypePairSet><T0,T1> represents a set of ordered pairs. Each
190ordered pair has a left hand side of type I<T0> and a right hand side
191of type I<T1>.
192
193B<FLSorted2TypePairSet> is intended for sets of ordered pairs which
194are compiled in advance, rather than incrementally added to. Currently
195no methods are provided for incremental addition and even if they
196are added in the future, they will not be fast. See L<"implementation">.
197
198=head1 IMPLEMENTATION
199
200B<FLSorted2TypePairSet> is implemented with sorted lists which are
201very memory efficient and reasonably fast for access, but probably slower
202than a good hash implementation. Sorted lists are not well suited
203for incremental addition of data.
204
205=head1 TERMINOLOGY
206
207Defining a few terms here will simplify the description and make the method
208arguments easier to understand.
209
210=over 4
211
212=item left hand side, or "lhs"
213
214The item appearing in the left hand side of an order pair. The
215left hand side of (a, b) is "a".
216
217=item right hand side, or "rhs"
218
219The item appearing in the right hand side of an order pair. The
220right hand side of (a, b) is "b".
221
222=item leftHandSideList
223
224I<leftHandSideList> is a list (without duplicates) of all the items
225appearing in the left hand side of an ordered pair.
226
227=item correspondingRightHandSideList
228
229I<correspondingRightHandSideList> is a list in the I<same> order
230as I<leftHandSideList>. Each element of I<correspondingRightHandSideList>
231is itself a list, containing (without duplicates) all of the
232right hand side items of pairs with the given left hand side.
233
234=back
235
236Together, leftHandSideList[i] and correspondingRightHandSideList[i]
237contain the information for all ordered pairs starting with the
238item "leftHandSideList[i]".
239
240=head2 Example
241
242Given set { (a,b), (a,c), (b,c) }:
243
244 The leftHandSideList of lhs is (a,b)
245
246 The correspondingRightHandSideList of "a" is:
247 correspondingRightHandSideList[0] -> (b,c)
248
249 The correspondingRightHandSideList of "b" is:
250 correspondingRightHandSideList[1] -> (c)
251

282=head1 INPUT TEXT FORMAT
283
284By default, the input file is in tab separated fields:
285
286 lhs rhs [rhs]...
287 lhs rhs [rhs]...
288
289The first field is a left hand side item. The second and
290following fields are corresponding right hand side items.

292
293 a b c
294 b c
295
