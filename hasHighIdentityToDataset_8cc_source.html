38void hasHighIdentityToDataset(){
39
40 /* ** declare and read in sequences ** */
41 FastaSeqSlurper seqSlurper;
42
43 const std::vector<LabeledSequence>
44 datasetSeqs( seqSlurper.slurpLabeledSequences( *arg_fastaIstreamPtr1 ) );
45
46 const std::vector<LabeledSequence>
47 querySeqs( seqSlurper.slurpLabeledSequences( *arg_fastaIstreamPtr2 ) );
48
49
50 const AminoScoreIdentity aminoScoreIdentity( aminoResidueIndexMap );
51
52 for( size_t querySeqIdx = 0; querySeqIdx < querySeqs.size() - 1; ++querySeqIdx ){
53
54 const LabeledSequence& querySeq = querySeqs.at(querySeqIdx);
55
56 if( arg_progressToStderr ) std::cerr << querySeq.name() << std::endl;
57
58 SeqGlobalAffineAligner aligner( aminoScoreIdentity, querySeq.residueIndices() );
59
60
61 bool queryMatches = false;
62
63 for( size_t datasetSeqIdx = querySeqIdx + 1;
64 datasetSeqIdx < datasetSeqs.size();
65 ++datasetSeqIdx ){
66
67 const LabeledSequence& datasetSeq = datasetSeqs.at(datasetSeqIdx);
68
69 FLEArray<alignScoreT> suffixBounds =
70 SeqGlobalAffineAligner::bestPossibleSuffixScore( aminoScoreIdentity,
71 datasetSeq.residueIndices() );
72
73 /* ** compute minimum alignment score for edge in cluster graph
74 * based on formula:
75 * adjusted identity = (# matches + adjust) / (average length + adjust)
76 */
77 const double maxLength = std::max( querySeq.size(), datasetSeq.size() );
78
79 const double& af = arg_lengthAdjustmentFactor;
80
81 GDB_ASSERTF( af < maxLength,
82 "lengthAdjustmentFactor = %g, too big for averageLength = %g",
83 arg_lengthAdjustmentFactor, maxLength );
84
85 double minScore
86 = arg_identityThreshold * ( af + maxLength ) - af;
87
88 // compensate for fact that SeqGlobalAffineAligner currently penalizes terminal gaps
89 // would be better to modify SeqGlobalAffineAligner...
90 if( querySeq.length() != datasetSeq.length() ){
91 minScore += abs( querySeq.length() - datasetSeq.length() )
92 * aminoScoreIdentity.gapExtension();
93 }
94
95 const double boundedScore
96 = aligner.scoreBounded( datasetSeq.residueIndices(),
97 suffixBounds,
98 (alignScoreT) minScore );
99
100 if( boundedScore > minScore ){
101
102 // print query sequence name if first match
103 if( !queryMatches ) std::cout << querySeq.name();
104
105 queryMatches = true;
106
107 // print match
108 std::cout << "\t" << datasetSeq.name();
109
110 // done, unless all similar pairs are to be output
111 if( !arg_outputAllPairs ){
112 goto NEXT_QUERY_SEQ;
113 }
114 }
115
116 } // end for datasetSeqIdx
117
118 NEXT_QUERY_SEQ:;
119 if( queryMatches ) std::cout << std::endl;
120 } // end for querySeqIdx
121
122} // end function hasHighIdentityToDataset
123
124} // end namescape cbrc
