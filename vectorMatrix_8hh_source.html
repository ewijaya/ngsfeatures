23 template <typename T, size_t numCols>
24 std::vector< std::vector<T> > fromCMatrix( T cmatrix[][numCols],
25 const size_t numRows ){
26 std::vector< std::vector<T> > retval;
27 for( size_t i = 0; i < numRows; ++i ){
28 size_t offset = numCols * i;
29 std::vector<T> curRow( *cmatrix + offset, *cmatrix + offset + numCols );
30 retval.push_back( curRow );
31 }
32 return retval;
33 }

55 std::vector< std::vector<T> > retVal;
56 size_t s;

61 retVal[i].resize(s);

69 // true if size of columns are all equal.
70 bool isRectangular( const std::vector< std::vector<T> >& m ){
71 if( !m.size() ) return true;
72 size_t numCols = m[0].size();
73 for( size_t i = 1; i < m.size(); ++i ){
74 if( m[i].size() != numCols ) return false;
75 }
76 return true;
