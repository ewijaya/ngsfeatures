14 * General Description: See header files.
15 *
16 * Definitions: dpTable(i,j) is the minimum cost alignment of prefixes that aligns s0[i] with s0[j].
17 * gpTable(i,j) is the minimum cost alignment of prefixes that ends in a "gap".
18 *
19 * In addition to gaps made by extending existing gaps, a gap may be a length 0 "initiator gap".
20 * an initiator gap consumes no characters but costs gapInitiation.
21 *
22 * thus gpTable(i,j) <= dpTable(i,j) + gapInitiation, because an initiator gap can be added
23 * to the alignment represented by dpTable(i,j).
24 */

31alignDistT GlobalAffineDist::dist( const FLArray<residueIndex>& s1 ){
32 alignDistT adjustedDistTMax = alignDistTMax - ad.maxDistance(); // to prevent overflow when adding a term before comparing
33
34 dpTable(0,0) = 0;
35 gpTable(0,0) = ad.gapInitiation();
36
37 // *debug* std::cout << "so.size = " << s0.size() << std::endl;
38 for( size_t j = 1; j <= s0.size(); ++j ){
39 dpTable(0,j) = adjustedDistTMax;
40 gpTable(0,j) = gpTable(0,j-1) + ad.gapExtension();
41 }
42
43 dpTable(1,0) = adjustedDistTMax;
44 gpTable(1,0) = gpTable(0,0) + ad.gapExtension();
45
46 size_t i = 0; // initialized to suppress compiler warning
47 for( size_t index0 = 1; index0 <= s1.size(); ++index0 ){
48 i = index0 % 2;
49 dpTable( i, 0 ) = adjustedDistTMax;
50 gpTable( i, 0 ) = gpTable( !i, 0 ) + ad.gapExtension();
51 for( size_t j = 1; j <= s0.size(); ++j ){
52 dpTable(i,j) = min( dpTable(!i,j-1), gpTable(!i,j-1) )
53 + ad.distance( s0[j-1], s1[index0-1] );
54 gpTable(i,j) = min( dpTable(i,j) + ad.gapInitiation(),
55 gpTable(i,j-1) + ad.gapExtension(),
56 gpTable(!i,j) + ad.gapExtension() );
57 // *debug* std::cout << dpTable( i, j ) << " ";
58 }
59 }
60 alignDistT dist = dpTable( i, s0.size() );
61 if( dist > gpTable( i, s0.size() ) ) dist = gpTable( i, s0.size() );
62 return dist;
63}

170 dpTable(0,0) = 0;
171 gpTable(0,0) = 0;
172
173 // allow a prefix of one of the two sequences to be skipped for free.
174 const alignDistT firstGapInitiation = 0;
175 const alignDistT firstGapPenalty = 0;
176
177
178 // *debug* std::cout << "so.size = " << s0.size() << std::endl;
179 for( size_t i = 1; i <= s0.size(); ++i ){
180 dpTable(0,i) = dpTable(0,i-1) + firstGapPenalty;
181 gpTable(0,i) = dpTable(0,i) + firstGapInitiation;
182 // *debug* std::cout << dpTable(0,i) << " ";
183 }
184 // *debug* std::cout << "\n";
185
186 dpTable(1,0) = firstGapPenalty;
187 gpTable(1,0) = dpTable(1,0);
188
189 size_t i = 0; // initialized to suppress compiler warning
190 for( size_t index0 = 1; index0 <= s1.size(); ++index0 ){
191 i = index0 % 2;
192 dpTable( i, 0 ) = dpTable( !i, 0 );
193 gpTable( i, 0 ) = dpTable( i, 0 );
194 // *debug* std::cout << dpTable( i, 0 ) << " ";
195 for( size_t j = 1; j <= s0.size(); ++j ){
196 dpTable(i,j) = min( dpTable(!i,j-1) + ad.distance( s0[j-1], s1[index0-1] ),
197 gpTable(i,j-1) + ad.gapExtension(),
198 gpTable(!i,j) + ad.gapExtension() );
199
200 gpTable(i,j) = min( dpTable(i,j) + ad.gapInitiation(),
201 gpTable(i,j-1) + ad.gapExtension(),

203 // *debug* std::cout << dpTable( i, j ) << " ";
204 }
205 // *debug* std::cout << std::endl;
206 }
207 alignDistT dist = dpTable( i, s0.size() );
208 if( dist > gpTable( i, s0.size() ) ){
209 dist = gpTable( i, s0.size() );

216 dpTable(0,0) = 0;
217 gpTable(0,0) = ad.gapInitiation();
218
219 // *debug* std::cout << "so.size = " << s0.size() << std::endl;
220 for( size_t i = 1; i <= s0.size(); ++i ){
221 dpTable(0,i) = dpTable(0,i-1) + ad.gapExtension();
222 gpTable(0,i) = dpTable(0,i) + ad.gapInitiation();
223 // *debug* std::cout << dpTable(0,i) << " ";
224 }
225 // *debug* std::cout << "\n";
226
227 dpTable(1,0) = ad.gapExtension();
228 gpTable(1,0) = dpTable(1,0) + ad.gapInitiation();
229
230 size_t i = 0; // initialized to suppress compiler warning
231 for( size_t index0 = 1; index0 <= s1.size(); ++index0 ){
232 i = index0 % 2;
233 dpTable( i, 0 ) = dpTable( !i, 0 ) + ad.gapExtension();
234 gpTable( i, 0 ) = dpTable( i, 0 ) + ad.gapInitiation();
235 // *debug* std::cout << dpTable( i, 0 ) << " ";
236
237 alignDistT minDistance = gpTable(i,0);
238 for( size_t j = 1; j <= s0.size(); ++j ){
239 dpTable(i,j) = min( dpTable(!i,j-1) + ad.distance( s0[j-1], s1[index0-1] ),
240 gpTable(i,j-1) + ad.gapExtension(),
241 gpTable(!i,j) + ad.gapExtension() );
242 if( dpTable(i,j) < minDistance ) minDistance = dpTable(i,j);
243 gpTable(i,j) = min( dpTable(i,j) + ad.gapInitiation(),
244 gpTable(i,j-1) + ad.gapExtension(),
245 gpTable(!i,j) + ad.gapExtension() );
246 if( gpTable(i,j) < minDistance ) minDistance = gpTable(i,j);
247 // *debug* std::cout << dpTable( i, j ) << " ";
248 }
249 if( minDistance > maxDistance ) return alignDistTMax;
250 // *debug* std::cout << std::endl;
251 }
252 alignDistT dist = dpTable( i, s0.size() );
253 if( dist > gpTable( i, s0.size() ) ){
254 dist = gpTable( i, s0.size() -1 );
255 }

260alignDistT GlobalAffineDist::feasibleBound( const FLArray<residueIndex>& s1 ){
261 alignDistT retVal = 0;
262 if( s0.size() == s1.size() ){
263 for( size_t i = 0; i < s0.size(); ++i ) retVal += ad.distance( s0[i], s1[i] ); // sum of diagonal
264 return retVal;
265 }
266 alignDistT curDist = 0;
267 if( s0.size() < s1.size() ){
268 for( size_t i = 0; i < s0.size(); ++i ) curDist += ad.distance( s0[i], s1[i] );
269 curDist += ad.gapInitiation();
270 curDist = curDist + ( s1.size() - s0.size() ) * ad.gapExtension(); // distance with all gaps at end of s0
271 retVal = curDist;
272 size_t s0last = s0.size() - 1;
273 size_t s1last = s1.size() - 1;
274 for( size_t i = 0; i < s0.size(); ++i ){
275 if( s1last-i < 0 ) break;
276 curDist -= ad.distance( s0[s0last-i], s1[s0last-i] );
277 curDist += ad.distance( s0[s0last-i], s1[s1last-i] );
278 if( curDist < retVal ) retVal = curDist;
279 }
280 return retVal;
281 }
282 // s0.size() > s1.size()
283 for( size_t i = 0; i < s1.size(); ++i ) curDist += ad.distance( s1[i], s0[i] );
284 curDist += ad.gapInitiation();
285 curDist = curDist + ( s0.size() - s1.size() ) * ad.gapExtension(); // distance with all gaps at end of s1
286 retVal = curDist;
287 size_t s1last = s1.size() - 1;
288 size_t s0last = s0.size() - 1;
289 for( size_t i = 0; i < s1.size(); ++i ){
290 if( s0last-i < 0 ) break;
291 curDist -= ad.distance( s1[s1last-i], s0[s1last-i] );
292 curDist += ad.distance( s1[s1last-i], s0[s0last-i] );
293 if( curDist < retVal ) retVal = curDist;
294 }
295 return retVal;
296}

303 dpTable(0,0) = 0;

306 alignDistT lowerBound = bestPossibleDist( ad, s1 );
307 if( lowerBound < bestPossibleDistVar ) lowerBound = bestPossibleDistVar;
308 int maxStepsFromDiag;
309 if( lowerBound > maxDistance ) return alignDistTMax;
310 maxStepsFromDiag = (maxDistance - lowerBound) / ad.gapExtension();
311 // std::cout << "maxStepsFromDiag:" << maxStepsFromDiag << std::endl;
312
313 for( size_t j = 1; j <= s0.size(); ++j ){
314 dpTable(0,j) = adjustedDistTMax;

316 // *debug* std::cout << dpTable(0,i) << " ";

321 gpTable(1,0) = gpTable(0,0) + ad.gapExtension();
322
323 size_t i = 0; // initialized to suppress compiler warning
324
325 int lengthDiff = s0.size() - s1.size();
326 int jStart;
327 int jEnd = 0; // initialized to suppress compiler warning
328
329 // *debug* std::cout << std::endl;

331
332 alignDistT minValueThisRow = 0;
333 for( size_t index0 = 1; index0 <= s1.size(); ++index0 ){
334 i = index0 % 2;
335 dpTable( i, 0 ) = adjustedDistTMax;
336 gpTable( i, 0 ) = gpTable( !i, 0 ) + ad.gapExtension();
337
338 // using minValueThisRow value from previous iteration
339 assert( minValueThisRow < alignDistTMax );
340 alignDistT newLowerBound = minValueThisRow + s1BestPossibleSuffixDists[index0-1];
341 if( newLowerBound > lowerBound ){
342 lowerBound = newLowerBound;
343
344 if( lowerBound > maxDistance ){
345 // std::cout << " minValueThisRow: " << minValueThisRow << " s1BestPossibleSuffixDists[" << index0-1 << "] = " << s1BestPossibleSuffixDists[index0-1] << std::endl;
346 // std::cout << "returning alignDistTMax\n";
347 return alignDistTMax;

356 // std::cout << "jStart, jEnd are: " << jStart << ", " << jEnd;
