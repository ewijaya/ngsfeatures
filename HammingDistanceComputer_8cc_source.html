20 const size_t& len1 = seq1.length();
21
22
23 const size_t shorterLen = std::min( len0, len1 );
24 size_t numMismatchesSeenSoFar = std::max( len0, len1 ) - shorterLen;
25

80size_t HammingDistanceComputer
81::computeBoundedGapAtBeg( const LabeledSequence& seq0,
82 const LabeledSequence& seq1,
83 const double& threshold
84 ){
85
86 const size_t& len0 = seq0.length();
87 const size_t& len1 = seq1.length();
88
89
90 const size_t shorterLen = std::min( len0, len1 );
91 const size_t longerLen = std::max( len0, len1 );
92 const size_t lenDiff = longerLen - shorterLen;
93
94 size_t numMismatchesSeenSoFar = lenDiff;
95
96 // if lengths are more different than THRESHOLD, give up immediately.
97 if( numMismatchesSeenSoFar >= threshold ){
98 return numMismatchesSeenSoFar;
99 }
100
101 const ResidueIndexMap::arrayT& querySeq = seq0.residueIndices();
102 const ResidueIndexMap::arrayT& curSeq = seq1.residueIndices();
103
104 ResidueIndexMap::arrayT::const_iterator queryItr;
105 ResidueIndexMap::arrayT::const_iterator curSeqItr;
106
107 if( len0 > len1 ){
108 queryItr = querySeq.constBegin();
109 curSeqItr = curSeq.constBegin() + lenDiff;
110 }else{
111 queryItr = querySeq.constBegin() + lenDiff;
112 curSeqItr = curSeq.constBegin();
113 }
114
115 for( /* no init */; queryItr != querySeq.constEnd(); ++queryItr, ++curSeqItr ){
116 if( *queryItr != *curSeqItr ){
117 if( ++numMismatchesSeenSoFar >= threshold ){
118 return numMismatchesSeenSoFar;
119 }
120 }
121 }
122
123 GDB_ASSERTF( curSeqItr == curSeq.constEnd() );
124
125 return numMismatchesSeenSoFar;

144
145 // if lengths are more different than THRESHOLD, give up immediately.

154 shorterPtr = &seq0;
155 longerPtr = &seq1;

157 shorterPtr = &seq1;
158 longerPtr = &seq0;
159 }
160
161 std::string::const_iterator shorterItr = shorterPtr->begin();
162 std::string::const_iterator longerItr = longerPtr->begin() + lenDiff;
163
164 std::string::const_iterator shorterEnd = shorterPtr->end();
165
166 for( /* no init */; shorterItr != shorterEnd; ++shorterItr, ++longerItr ){
167 if( *shorterItr != *longerItr ){
168 if( ++numMismatchesSeenSoFar >= threshold ){
169 return numMismatchesSeenSoFar;
170 }
171 }
172 }
173
174 GDB_ASSERTF( longerItr == longerPtr->end(), "end of sequences did not match" );
175
176 return numMismatchesSeenSoFar;
