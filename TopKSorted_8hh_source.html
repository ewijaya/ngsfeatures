8 * Purpose: Compute the top k items in sorted order for a series of numbers that may be too large
9 * to fit in memory.
10 *
11 * Example: To find the top 10 doubles of a large set of numbers using no more than 1000 doubles
12 * worth of memory do the following:
13 *
14 * TopKSorted topK<double>( 10, 1000 );
15 * // now topK.getK() == 10
16 * for each double value val{
17 * topK.insertValueIfInTopK( val );
18 * }
19 * double* sortedList = topK.getSortedList();
20 * for( size_t i = 0; i < topK.getK(); ++i ){
21 * std::cout << sortedList[i] << " ";
22 * }
23 *
24 * Caveat: insertValueIfInTopK(T) must be called at least k times for getSortedList() to return
25 * a valid list.
26 */

41 TopKSorted( const size_t& k, const size_t& maxStackSize )

54 }

64
65 const T& operator()( const size_t& i ) const{
66 assert( i < k() );

70
71
72 // returns true when VAL cannot possibly in the top k.
73 bool cannotBeInTopK( const T& val ) const{
74
75 if( top < k() ){
76 return false; // have not even seen k values yet!
77 }
78
79 return cmp( minAcceptableVal, val );
80 }
81
82
83 void insertValueIfInTopK( const T& val ){
84 // if < k elements in stack, insert val unconditionally.
85 if( top < k() ){
86 if( top == 0 ) minAcceptableVal = val;
87 if( cmp( minAcceptableVal, val ) ) minAcceptableVal = val;
88 _stack[top++] = val;
89 return; // ** method exit point **
90 }
91
92 // stack already has â‰§ k elements
93 if( !cmp( val, minAcceptableVal ) ){
94 return; // ** method exit point **
95 }
96
97 // val is in top k so far.
98 if( top < maxStackSize ){
99 _stack[top++] = val;
100 return; // ** method exit point **
101 }
102
103 // assert( top == maxStackSize ), also top >= k.
104 // val should be added but stack is full, so sort stack to put the top k items at the front.
105 std::nth_element( _stack.begin(), _stack.begin() + k()-1, _stack.begin() + top, cmp );
106 minAcceptableVal = _stack[ k()-1 ];
107 top = k();
108 insertValueIfInTopK( val ); /* Achtung! Recursive call. */
109 }
110
111
112 void sort(){ // sort the top k elements.
113 std::partial_sort( _stack.begin(),
114 _stack.begin() + std::min( k(), top ), _stack.end(),
115 cmp );
116 }
117
118 void clear(){ top=0; }
119
120 TopKSorted<T, BinaryPredicate> operator=( const TopKSorted<T, BinaryPredicate>& topK ){
121 TopKSorted<T, BinaryPredicate> retVal( topK );
122 return retVal;
123 }
124private:
125 size_t top;
126 const size_t _k;
127 const size_t maxStackSize;
128 FLEArray<T> _stack;
129 T minAcceptableVal;
130 BinaryPredicate cmp;
131};
