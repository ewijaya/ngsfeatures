30 inline ssize_t getline( char **lineptr, size_t *n, FILE *ifp ){
31#endif // JUST_TESTING_GETLINE
32 size_t& bufferSize = *n;
33 char*& buffer = *lineptr;
34 if( buffer == NULL ){ // buffer has not been pre-allocated by caller.
35 bufferSize = 256;
36 buffer = (char*) malloc( bufferSize );
37 assert( buffer );
38 }
39 ssize_t numBytesRead = 0;
40 const char notOverwrittenMark = 'A'; // anything by '\0' should do.
41 char* bufferCurPos = buffer;
42 buffer[ bufferSize - 1 ] = notOverwrittenMark;
43 char* fgetsRetVal = fgets( bufferCurPos, bufferSize, ifp );
44 if( fgetsRetVal == NULL ){ // no characters read from ifp
45 *lineptr = NULL;
46 return -1;
47 }
48
49 numBytesRead += strlen( bufferCurPos );
50 if( buffer[ bufferSize -1 ] == notOverwrittenMark ){ // line fit into original buffer.
51 return numBytesRead;
52 }
53
54 for( ;; ){ // keep reallocating with doubled size until the line fits.
55 bufferSize *= 2;
56 assert( (buffer = (char*) realloc( buffer, bufferSize )) );
57 buffer[ bufferSize - 1 ] = notOverwrittenMark;
58 bufferCurPos = buffer + numBytesRead;
59 if( fgets( bufferCurPos, bufferSize - numBytesRead, ifp ) == NULL ) break;
60 numBytesRead += strlen( bufferCurPos );
61 if( buffer[ bufferSize -1 ] == notOverwrittenMark ) break;
62 }
63 return numBytesRead;
64 }

66}
