104 static alignScoreT singleSeqBasedScoreUpperBound( const AminoScore& resSubScore,
105 const ResidueIndexMap::arrayT& s1 ){
106 alignScoreT score = 0;
107 for( size_t i = 0; i < s1.size(); ++i ) score += resSubScore.maxScore( s1[i] );
108 return score;
109 }
