48void KnapsackEnumerator::
49computeCombinationsAux( const size_t& dfsLevel ){
50
51 // Save current combination
52 _combinations.push_back( idStack );
53 _combinations[ _combinations.size() - 1 ].resize( dfsLevel+1 );
54
55
56 /* ***** Continue Depth First Search at next level ***** */
57
58 const idT nextLevel = dfsLevel + 1;
59 const idT& prevId = idStack[dfsLevel];
60 const weightT& curSum = sumStack[dfsLevel];
61
62
63 for( size_t id = prevId + 1; id < objectUniverse().size(); ++id ){
64
65 const weightT newSum = curSum + objectUniverse()( id ).weight();
66
67 if( newSum <= capacity() ){
68 idStack [ nextLevel ] = id;
69 sumStack[ nextLevel ] = newSum;
70 computeCombinationsAux( nextLevel );
71 }else{
