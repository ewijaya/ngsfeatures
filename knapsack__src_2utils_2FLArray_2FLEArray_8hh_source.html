507template <typename T>
508inline void FLEArray<T>::setFromLiteral( const std::string semanticTypeDescriptor,
509 const std::string arrayLiteral ){
510
511 if( semanticTypeDescriptor == "boolean" ){
512 resize( arrayLiteral.size() );
513 for( size_t i = 0; i < size(); ++i ){
514 switch( arrayLiteral[i] ){
515 case '0':
516 (*this)[i] = 0;
517 break;
518 case '1':
519 (*this)[i] = 1;
520 break;
521 default:
522 std::cout << __FILE__ << ":" << __LINE__
523 << " Error: element of arrayLiteral not equal to either '0' nor '1'" << std::endl;
524 exit( -1 );
525 } // end switch arrayLiteral[i]
526 } // end for i, (*this)[i]
527 }
528 else if( semanticTypeDescriptor == "numeric" ){
529 std::vector<std::string> valStrings
530 = perlish::split( boost::regex( ",\\s*" ), arrayLiteral );
531
532 resize( valStrings.size() );
533 for( size_t i = 0; i < valStrings.size(); ++i ){
534 a[i] = boost::lexical_cast<T>( valStrings[i] );
535 }
536 }
537 else{
538 GDB_DIEF( "unsupported semantic type \"%s\"", semanticTypeDescriptor.c_str() );
539 }
540
541} // end FLEArray<T>::setFromLiteral( const std::string&, const std::string& arrayLiteral )

584template <typename T>
585void FLEArray<T>::resize( const size_t& sz ){
586
587 if( a ){ // if array already initialized
588
589 if( s == sz ) return;
590
591 T* b = new T[sz];
592 assert( b );
593
594 const size_t numElemsToCopy = ( s < sz ) ? s : sz;
595 memcpy( b, a, numElemsToCopy*sizeof(T) );
596
597 delete []a;
598 a = b;
599 s = sz;
600 }
601 else{ // if array is uninitialized
602 setSize( sz );
603 }
604}

608template <typename T>
609FLEArray<T> FLEArray<T>::operator=( const FLEArray& fla ){
610
611 // die with error if fla is a non-initialized array
612 if( !fla.a ){
613 if( fla.size() != 0 ){
614 std::cout << __FILE__ << ":" << __LINE__ << " Error: tried to assign from null array with non-zero size\n";
615#ifdef GDB_DEBUG
616 for( ;; );
617#endif // defined GDB_DEBUG
618 exit( -1 );
619 }
620 s = 0;
621 a = NULL;
622 }
623
624 // die with error if array sizes don't match
625 if( a ){
626 if( size() != fla.size() ){
627 std::cout << __FILE__ << ":" << __LINE__ << " Error: tried to assign array of size " << fla.size() << " to one of size: " << size() << std::endl;
628#ifdef GDB_DEBUG
629 for( ;; );
630#endif // defined GDB_DEBUG
631 exit( -1 );
632 }
633 }
634
635 // copy the contents of fla and return *this
636 assign( fla.begin(), fla.end() );
637 return *this;
638}

691template <typename T>
692size_t FLEArray<T>::find( const T& val, const size_t& begIdx, const size_t& endIdx ){
693 // validate arguments
694 assert( _validRange( begIdx, endIdx ) );
695
696 // fill in value
697 const T* const endPtr = a+endIdx;
698
699 for( const T* curPtr = a+begIdx; curPtr < endPtr; ++curPtr ){
700 if( *curPtr == val ){
701 return( curPtr - a );
702 }
703 }
704 return npos;

949=head1 SYNOPSIS
950
951Array to hold fixed byte length elements. Single memory region, no dynamic size change.
952
953
954=head1 DESCRIPTION
955
956=head2 FLEArray (Base Class)
957
958The B<FLEArray> family of classes can correctly hold fixed length elements. It
959is *B<not>* appropriate for variable length type elements or pointer containing types
960such as C<std::string> in which the pointed to material is also considered as part
961of the object.
962
963The B<FLArray> family of classes is provided
964for variable length types such as C<std::string>, but at this point I
965would recomment using L<std::vector|FLEArray vs. std::vector> for such types.
966
967=head2 FLEArrayFast (Derived Class)
968
969B<FLEArrayFast> is a derived class of B<FLEArray> which does not normally
970L<check that indices are in range|Index Range Checking>. B<FLEArrayFast> is B<!DEPRECATED!>,
971instead use B<FLEArray> and set B<NDEBUG> to turn of range checking.
972
973=head2 FLENumArray (Derived Class)
974
975B<FLENumArrayE<lt>TE<gt>> is an extension of B<FLEArray> which assumes element type B<T> can do arithmetic. It supplies methods such as C<sum()> which returns the summation of the elements.
976B<FLENumArrayFast> is exactly analogous to B<FLEArrayFast>.
977
978=head2 Ranges
979
980There is no "Range" class defined but several methods take index regions in the form
981C<size_t begIdx, size_t endIdx>. As in the C++ standard template library, these intervals
982are half open, covering indices I<i>: I<begIdx> I<i> < I<endIdx>. In this documentation
983this is often written as S<"[begIdx, endIdx)">.
984The empty range is represented by setting I<begIdx> and I<endIdx> to
985the same value (any value is fine for this).

1269=head1 RELATED CLASSES
1270
1271The B<FLArray> family of classes can hold objects (C<std::string>) which cannot
1272be copied with C<memcpy> or compared with C<memcmp>. Currently there is (even)
1273less reason to use B<FLArray> instead of L<std::vector|FLEArray vs. std::vector>.
1274
1275The B<MallocArrayE<lt>TE<gt>> family of classes further restricts the element type I<T>
1276to not require a constructor call upon creation, in fact it can hold user defined element
1277types that do not define a null constructor. This allows C<malloc> to be used
1278instead of C++ style array allocation (C<new T[size]>). The only place where this
1279currently seems to give a potential advantage is in the L<resize|SIZE RELATED METHODS>
1280method, where the use of C<realloc> potentially avoids copying elements to a new
1281area of memory.
1282
1283
1284=head1 FLEArray vs. std::vector
1285
1286B<FLEArray> is largely redundant to C<std::vector> but a few methods utilize
1287the assumption of fixed element size to advantage by using highly optimized
1288byte based library calls such as C<memcpy> and C<memcmp>. In principle C<std::vector> could
1289do this for specialized classes such as C<std::vectorE<lt>intE<gt>>, but I do not think
1290it usually does -- and in any case this would only work for the types which
1291happened to be specialized.
1292
1293The assumption of fixed length non-pointer element type allows B<FLEArray> to
1294provide simple and efficient L<binary I/O|"BINARY I/O"> methods.
1295
1296Another significant difference between B<FLEArray> and C<std::vector> is that
1297B<FLEArray> does not have the notion of capacity (nor the C<size_t> size overhead)
1298that C<std::vector> has; B<FLEArray> never changes size automatically,
1299it must be L<resized|SIZE RELATED METHODS> explicitly.
