25template<typename T, typename BinaryPredicate = std::greater<T> >
26class TopKUnique{
27public:
28 TopKUnique( size_t k ){ assert( k > 0 ); this->k = k; }
29 void insert( T val ){
30 if( size() < k ){
31 s.insert( val );
32 if( size() == k ) minAcceptableVal = *( minPos() );
33 return;
34 }
35 if( !cmp( val, minAcceptableVal ) ) return;
36 s.insert( val );
37 if( size() > k ){
38 s.erase( minPos() );
39 minAcceptableVal = *( minPos() );
40 }
41 }
42 T maxVal() const{ return *s.begin(); }
43 T minVal() const{ return *( minPos() ); }
44 const std::set<T, BinaryPredicate>& getSet() const{ return s; }
45 size_t size() const{ return s.size(); }
46private:
47 std::set<T, BinaryPredicate>::iterator minPos() const{
48 std::set<T, BinaryPredicate>::iterator last = s.end();
49 return( --last );
50 }
51 size_t k;
52 T minAcceptableVal;
53 BinaryPredicate cmp;
54 std::set<T, BinaryPredicate> s;
55};
