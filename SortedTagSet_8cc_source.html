31void SortedTagSet::readFromTextStream( std::istream& iStream ){
32
33 std::string line;
34
35 DO_OR_DIEF( perlish::slurpLine( line, iStream ),
36 "Input error. Empty tag list stream?" );
37
38 size_t numTags;
39
40 perlish::fromString( numTags, line );
41
42
43 /* ***** Initialize Vector and Trim Size ***** */
44 ResidueIndexMap::vectorT residueIndices;
45
46 v.resize( numTags );
47
48
49 for( size_t tagCount = 0; tagCount < numTags; ++tagCount ){
50
51 DO_OR_DIEF( perlish::slurpLine( line, iStream ),
52 "End of file after only %zu out of %zu total tags read",
53 tagCount, numTags );
54
55 residueIndexMap().assignResidueIndices( residueIndices, line );
56
57 v[tagCount] = residueIndices;
58
59 }
60
61
62 DO_OR_DIEF( !perlish::slurpLine( line, iStream ),
63 "Extra unexpected line: \"%s\" found after tags", line.c_str() );
64
65 assertIsSorted();
66} // end method readFromTextStream
