12 * Purpose: Provide a class similar to FLArray in that it does not autmotically expand -- but
13 * that provides a way to manually shrink or grow.
14 *
15 * Description: VLArray is a fixed length array that checks the unsigned int index for being
16 * too large when using the operator [].
17 *
18 * VLArrayFast is a derived class that avoids that small overhead by not checking
19 * the index passed to operator [] for being too large. However, when debugging
20 * FLARRAY_ALWAYS_CHECK_INDICES can be defined to force index checking with VLArrayFast too.
21 *
22 * FLNumArray is an extension of VLArray which assumes elemT can do arithmetic. It supplies
23 * methods such as sum() which returns the summation of the elements.
24 *
25 * FLNumArrayFast is exactly analogous to VLArrayFast. In fact I tried to have it multiple
26 * inherit from FLNumArray and VLArrayFast -- but I couldn't get that to work...
27 */

102 setSize( newSize );

231 FLNumArray operator=( const FLNumArray& fla ){
232 if( !fla.a ){
233 assert( !a ); // only allow empty array to be assigned to another empty array.
234 return (*this);
235 }
236 if( a ){
237 assert( size() == fla.size() );
238 }

240 s = fla.size();

244 return *this;
245 }
246 FLNumArray& operator*=( elemT el ){
247 for( size_t i = 0; i < size(); ++i ) a[i] *= el;
248 return( *this );
249 }
250 FLNumArray& operator/=( elemT el ){
251 for( size_t i = 0; i < size(); ++i ) a[i] /= el;
252 return( *this );
253 }
254 elemT sum() const{
255 elemT retVal = 0;
256 for( size_t i = 0; i < size(); ++i ) retVal += a[i];
257 return retVal;
258 }
259 elemT max() const{
260 assert( size() > 0 );
261 elemT retVal = a[0];
262 for( size_t i = 1; i < size(); ++i ) if( retVal < a[i] ) retVal = a[i];
263 return retVal;
264 }
265 size_t maxIndex() const{ // returns first index in case of ties.
266 assert( size() > 0 );
267 size_t retIndex = 0;
268 elemT maxVal = a[0];
269 for( size_t i = 1; i < size(); ++i ) if( maxVal < a[i] ){ maxVal = a[i]; retIndex = i; }
270 return retIndex;
271 }
272 void squareElements(){ for( size_t i = 0; i < size(); i++ ) a[i] = a[i] * a[i]; }
273};
