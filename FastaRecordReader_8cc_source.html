77 /* read next record from iStream and assign contents to _curRecord
78 * if successful, return pointer to curRecord, otherwise return NULL
79 */
80const FastaRecord*
81FastaRecordReader::nextRecord( std::istream& iStream ){
82
83 _curIstreamPtr = &iStream; // copy to global variable for convenience.
84
85 _curRecord.clear();
86
87 if( !readRecordHead() ) return NULL; // EXIT: no more records.
88
89 GDB_ASSERTF( curRecord().headLine().size(), "Program Error: unexpected empty head line" );
90
91 computeLabels();
92
93
94 /* ******* Read body of record ******* */
95 while( curLine_readToNextBodyNonCommentLine() ){
96
97 // append sequence characters from this line onto sequence.
98 std::remove_copy_if(
99 curLine().begin(), curLine().end(),
100 back_inserter( _curRecord._seq ),
101 std::not1( std::ptr_fun<int,int>(inputOptions().seqCharFilter()) )
102 );
103 }
104
105
106 // remove trailing '*' from seq if present
107 if( seq().at( seq().size() - 1 ) == fastafmt::seqTerminationMarker() ){
108 _curRecord._seq.resize( seq().size() - 1 ); // remove it
109 }
110
111 return &_curRecord;
112
113}

119/* attempt to read head of next record in curIstream
120 * return false if end of stream encountered
121 *
122 * comment lines encountered before the record
123 * start line are stored in _commentLinesBeforeHead;
124 *
125 * head line is stored in _headLineVerbatim
126 *
127 * Execution aborts with error, if non-comment line
128 * encountered before next record start.
129 */
130bool FastaRecordReader::readRecordHead(){
131
132 // read lines until EOF or marker start character seen
133 while( curLine_readNonHeadLine() ){
134 if( curLine_isComment() ){
135 _curRecord._commentLinesBeforeHead.push_back( curLine() );
136 }
137 GDB_DIEF( "Could not parse line: \"%s\"", curLine().c_str() );
138 }
139
140 if( curIstream().eof() ) return false; // EXIT: no more records in stream.
141
142 std::getline( curIstream(), _curLine );
143
144 _curRecord._headLine = curLine();
145
146
147 return true; // EXIT: record read.
148
149} // end readRecordHead( std::istream& )

153// compute curRecord().labels from head by splitting on delimiterRegex
154void FastaRecordReader::computeLabels(){
155
156 GDB_ASSERTF( !curRecord().labels().size(), "Programming error: expected labels empty" );
157
158 boost::smatch recordStartMatch;
159
160 DO_OR_DIEF( boost::regex_search( curRecord().headLine(),
161 recordStartMatch,
162 fastafmt::recordStartRegex() ),
163 "expected record start marker" );
164
165
166 _curRecord._labels =
167 perlish::split( inputOptions().betweenLabelsDelimiterRegex(),
168 curRecord().headLine().begin() + recordStartMatch.length(),
169 curRecord().headLine().end() );
170
171 if( !curRecord().labels().size() ){
172 _curRecord._labels.push_back( defaultName() );
173 }
174
175
176} // end computeLabels()

180bool FastaRecordReader::curLine_readToNextBodyNonCommentLine(){
181
182 while( curLine_readNonHeadLine() ){
183 _curRecord._bodyLines.push_back( curLine() );
184
185 if( !curLine_isComment() ) return true; // EXIT.
186 }
187
188 return false; // EXIT, end of record reached before non-comment line.
189}

193// if non-head line follows current position in curIstream,
194// assign it to global _curLine
195// otherwise do nothing and return false.
196bool FastaRecordReader::curLine_readNonHeadLine(){
197
198 if( curIstream().eof() ||
199 (curIstream().peek() == fastafmt::recordStartMarker()) ){
200 return false; // EXIT, at end of record.
201 }
202
203 std::getline( curIstream(), _curLine );
204
205 if( curIstream().fail() ){
206 return false; // necessary because getline() reads one empty line before generated an EOF
207 }
208
209 return true;
210}
