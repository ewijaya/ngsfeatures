48 inline seqTypeT guessSeqType( const std::string& seq ){
49 if( aminoResidueIndexMap.validResidues( seq ) ) return amino;
50 if( DNAProperResidueIndexMap.validResidues( seq ) ) return DNA;
51 if( RNAResidueIndexMap.validResidues( seq ) ) return RNA;
52 return unknown;
53 }

59inline const ResidueIndexMap& residueIndexMapGuessedFromResidues( const std::string& residues ){
60 if( DNAProperResidueIndexMap.validResidues( residues ) ) return DNAProperResidueIndexMap;
61 if( DNAWithNResidueIndexMap.validResidues( residues ) ) return DNAWithNResidueIndexMap;
62 if( RNAResidueIndexMap.validResidues( residues ) ) return RNAResidueIndexMap;
63 if( aminoResidueIndexMap.validResidues( residues ) ) return aminoResidueIndexMap;
64 std::cout << "Error: failed to guess sequence type:\n";
65 exit( -1 );
66}

69inline const ResidueIndexMap& residueIndexMapFromName( const std::string& seqTypeName ){
70 if( seqTypeName == "DNA" ) return DNAProperResidueIndexMap;
71 if( seqTypeName == "dna" ) return DNAProperResidueIndexMap;
72 if( seqTypeName == "RNA" ) return RNAResidueIndexMap;
73 if( seqTypeName == "rna" ) return RNAResidueIndexMap;
74 if( seqTypeName == "amino" ) return aminoResidueIndexMap;
75 std::cout << "Error: unknown sequence type name: \"" << seqTypeName << "\"\n";
