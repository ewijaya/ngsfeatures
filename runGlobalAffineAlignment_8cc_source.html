39 void runGlobalAffineAlignment(){
40 FastaRecordReader fr;
41 AminoScorePam120 aScore;
42
43 const FastaRecord* recordPtr = fr.nextRecordOrDie( *fastaIstreamPtr );
44
45 ResidueIndexMap::arrayT s1( aminoResidueIndexMap.toResidueIndicesVector( fr.seq() ) );
46
47 SeqGlobalAffineAligner gas( aScore, s1 );
48
49 recordPtr = fr.nextRecordOrDie( *fastaIstreamPtr );
50
51 ResidueIndexMap::arrayT s2( aminoResidueIndexMap.toResidueIndicesVector( fr.seq() ) );
52
53 Alignment al = gas.alignment( s2 );
54
55 printIfHtml( "<HTML><BODY>\n" );
56 std::cout << "alignment score: " << al.score( aScore ) << std::endl;
57 printIfHtml( "<BR>\n" );
58
59 if( outputHtml ){
60 if( markColumns ) al.outputHtmlMarkSame( std::cout, markDifferent );
61 else al.outputHtml ( std::cout );
62 }
63 else{ // output plain text
64 if( markColumns ) al.outputTextMarkSame( std::cout, markDifferent );
65 else std::cout << al;
66 }
67 printIfHtml( "</BODY></HTML>\n" );
68 }
69} // end namescape cbrc
