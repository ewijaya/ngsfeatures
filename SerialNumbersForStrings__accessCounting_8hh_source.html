128 // Get serial number of string, creating a new serial number if necessary.
129 // Increments access count of label s.
130 template <typename serialNumT>
131 serialNumT SerialNumbersForStrings_accessCounting<serialNumT>::getSerialNumber( const std::string& s ){
132 for( size_t i = 0; i < size(); ++i ){
133 if( s == elems[i].value() ){
134 elems[i].incrementAccessCount();
135 return i;
136 }
137 }
138 elems.push_back( s );
139 return( size() - 1 );
140 }
