20 size_t RecountNeighborProbGraphWriter::write( std::ofstream& ofStream,
21 std::istream& tagNeighborlistIstream ){
22
23
24 // declare vectors to hold neighbor data
25 std::vector<tagIdT> neighborIds;
26 std::vector<probT> neighborProbs;
27 RecountNeighborList neighborList;
28
29 size_t nodeCount = 0; // number of neighborhood's written.
30
31 // write signature
32 binaryIO::writeContentsOnly( RecountNeighborProbGraphFormat::signature(), ofStream );
33
34 // save ofstream pointer so we can go back and write it later
35 const std::ofstream::pos_type posBeforeNodeCount = ofStream.tellp();
36
37 // make room for writing nodeCount afterwards.
38 ofStream.seekp( sizeof(nodeCount), std::ios_base::cur );
39
40
41 for( ; ; ++nodeCount ){
42
43 const std::string line = perlish::slurpLine( tagNeighborlistIstream );
44
45 if( !line.size() ) break;
48 GDB_ASSERTF( !boost::regex_search( line, boost::regex("\t\t") ),
49 "Input error. Fields should be separted by exactly one tab character."\
50 " line:\n%s\n", line.c_str() );
51
52
53 const stringVecT fields = perlish::split( "\t", line );
54
55
56 GDB_ASSERTF( fields.size() > 3,
57 "Expected at least 3 fields, but got %zu", fields.size() );
58
59 GDB_ASSERTF( (fields.size() % 2),
60 "Expected an odd number of fields, but got %zu", fields.size() );
61
62 /* *** Pop off tag sequence and count *** */
63 const std::string tagSeq = fields[0];
64
65
66 { // ***** Build vector of neighborIds and neighborProbs
67
68 const neighborhoodSizeT listBegIdx = 1; // neighbor list start index
69
70 neighborIds .resize ( 0 );
71 neighborProbs.resize ( 0 );
72
73 /* ***** Parse FIELDS into id and probability lists ***** */
74 for( size_t i = listBegIdx; i < fields.size();
75 i++ /* also incremented in loop body */
76 ){
77
78
79 const std::string neighborSeq = fields[i];
80
81 GDB_ASSERTF( ( neighborSeq != tagSeq ),
82 "tagId:%s found in its own neighbor list", tagSeq.c_str() );
83
84 const TagSet::unpackedSeqT neighborUnpackedSeq
85 = tagId_to_seq.toUnpackedSeq( fields[i] );
86
87
88 // When tag present, this implementation does 2 binary searches.
89 // Could be optimized to 1.
90 if( tagId_to_seq.has( neighborUnpackedSeq ) ){
91 const size_t neighborId = tagId_to_seq( neighborUnpackedSeq );
92 neighborIds .push_back( neighborId );
93 neighborProbs.push_back( boost::lexical_cast<probT> ( fields[++i] ) );
94 }
95 else{ // skip neighbor whose tag is not in the tag list
96 // std::cerr << "skipping tag: " << neighborSeq << std::endl;
97 ++i; // skip over unused probability field.
98 }
99 }
100 } // END: Build vector of neighborIds and neighborProbs block.
101
102
103 { // ***** Check validity of probs and add self to neighbor list
104
105 probT sumOtherProbs = 0.0;
106 BOOST_FOREACH( const probT& prob, neighborProbs ){
107 GDB_ASSERTF( ( (prob >= 0) && (prob < 0.5) ),
108 "invalid probability value: %f", prob );
109 sumOtherProbs += prob;
110 }
111
112 /*--------------------------------------------------
113 * GDB_ASSERTF( ( (sumOtherProbs >= 0) && (sumOtherProbs < 0.9) ),
114 * "When processing tag \"%s\", sumOtherProbs of non-self neighbors should be in [0,0.9] but was: %g",
115 * tagSeq.c_str(), sumOtherProbs );
116 *--------------------------------------------------*/
117
118 if ( sumOtherProbs <= 1) {
119 // push self onto neighbor list
120 neighborIds .push_back( tagId_to_seq.getSerialNumberOrDie( tagSeq ) );
121 neighborProbs.push_back( 1.0 - sumOtherProbs );
122 }
123 }
124
125
126 /* ********** WRITE OUTPUT ********** */
127
128 neighborList.set( tagId_to_seq.getSerialNumberOrDie( tagSeq ),
129 neighborIds, neighborProbs );
130
131
132 neighborList.write( ofStream );
133
134
135 } // end for slurpLine( tagNeighborlistIstream )
136
137 // overwrite nodeCount at beginning of ofStream
138 ofStream.seekp( posBeforeNodeCount, std::ios_base::beg );
139 ofStream.write( (char*) &nodeCount, sizeof(nodeCount) );
140
141 return nodeCount;
142 }
143
144} // end namespace cbrc
