36double betacf(const double& a, const double& b, const double& x) {
37
38 double aa,c,d,del,h,qab,qam,qap;
39 const int MAXIT = 100;
40 const double FPMIN = 1.0e-300;
41
42 const double EPSS = 3.0e-16;
43
44 qab=a+b;
45 qap=a+1.0;
46 qam=a-1.0;
47 c=1.0;
48 d=1.0-qab*x/qap;
49 if(fabs(d) < FPMIN) d = FPMIN;
50
51 d=1.0/d;
52 h=d;
53
54
55 int m;
56 for(m = 1; m <= MAXIT; m++) {
57
58 int m2= 2 * m;
59
60 aa=(double)m*(b-(double)m)*x/((qam+(double)m2)*(a+(double)m2));
61
62 d=1.0+aa*d;
63 if(fabs(d) < FPMIN) d = FPMIN;
64 c=1.0+aa/c;
65 if(fabs(c) < FPMIN) c = FPMIN;
66 d=1.0/d;
67 h *= d*c;
68
69 aa = -(a+m)*(qab+m)*x/((a+m2)*(qap+m2));
70
71
72 d=1.0+aa*d;
73 if(fabs(d) < FPMIN) d = FPMIN;
74 c=1.0+aa/c;
75 if(fabs(c) < FPMIN) c = FPMIN;
76 d=1.0/d;
77 del=d*c;
78 h *= d*c;
79
80 if(fabs(del-1.0) < EPSS) break;
81 }
82 if(m > MAXIT) throw std::string("a or b too big, or MAXIT too small");
83 return h;
84
85}

88double betai(const double& a, const double& b, const double& x) {
89
90 double bt;
91 if(x < 0.0 || x > 1.0) {
92 throw std::string("bad x in betai.");
93 }
94 if(x == 0.0 || x == 1.0) bt = 0.0;
95 else
96 bt = exp(lgamma(a+b)-lgamma(a)-lgamma(b)+a*log(x)+b*(log(1.0-x)));
97
98
99 if(x < (a+1.0)/(a+b+2.0))
100 return bt*betacf(a,b,x)/a;
101 else
102 return 1.0-bt*betacf(b,a,1.0-x)/b;
103
104}

106double TtoP(const double& t_val, const double& dfree) {
107
108 if(dfree < 1 || isnan(t_val)) {
109 return std::numeric_limits<double>::signaling_NaN();
110 }
111 return betai(0.5 * dfree, 0.5, dfree/(dfree + t_val * t_val));
