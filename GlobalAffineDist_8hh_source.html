53 static alignDistT bestPossibleDist( const AminoDistance& ad, const FLArray<residueIndex>& s1 ){
54 alignDistT dist = 0;
55 for( size_t i = 0; i < s1.size(); ++i ) dist += ad.minDistance( s1[i] );

58 static FLArray<alignDistT> bestPossibleSuffixDist( const AminoDistance& ad, const FLArray<residueIndex>& s0 ){
59 FLArray<alignDistT> dists( s0.size() );
60 if( !s0.size() ) return dists;
61 dists[ s0.size()-1 ] = ad.minDistance( s0[ s0.size()-1 ] );
62 for( size_t i = s0.size()-2; i < s0.size() /* while i >= 0 */ ; --i ){
63 dists[ i ] = dists[ i+1 ] + ad.minDistance( s0[i] );
64 }
65 return dists;
66 }
