72 // after returning, the new y coordinate is the weighted average of
73 // the old y coordinate and the point p. The new weight is the
74 // sum of the old weight and the weight of p.
75 void averageYCoordinateAndSumWeightsWith( const WeightedPoint2D<XT, YT, WT>& p ){
76 double totalWeight = static_cast<double>( w + p.weight() );
77 yVar = (w * yVar + p.weight() * p.y()) / totalWeight;
78 w = totalWeight;
79 }
80
81 // after returning, new x,y coordinates are the weighted average of
82 // the old x,y coordinates and the point p. The new weight is the
83 // sum of the old weight and the weight of p.
84 void averageCoordinatesAndSumWeightsWith( const WeightedPoint2D<XT, YT, WT>& p ){
85 double totalWeight = static_cast<double>( w + p.weight() );
86 xVar = (w * xVar + p.weight() * p.x()) / totalWeight;
87 yVar = (w * yVar + p.weight() * p.y()) / totalWeight;
88 w = totalWeight;
89 }
