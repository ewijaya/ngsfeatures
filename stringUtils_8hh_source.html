70 /* replace all occurrences of oldSubstring in S with newSubstring
71 * replacement is down sequentially from the start of S.
72 * which determines what happens to overlapping occurences of oldSubstring.
73 *
74 * The number of replacements done is returned.
75 */
76 inline std::string::size_type
77 stringReplaceAll( std::string& s,
78 const std::string& oldSubstring,
79 const std::string& newSubstring ){
80 std::string::size_type curMatchStartPos = 0;
81 size_t numReplacementsDone = 0;
82
83 while( (curMatchStartPos = s.find( oldSubstring )) != std::string::npos ){
84 s.replace( curMatchStartPos, oldSubstring.size(), newSubstring );
85 ++numReplacementsDone;
86 curMatchStartPos += newSubstring.size();
87 }
88
89 return numReplacementsDone;
90 } // end stringReplaceAll
