65// read stringVector, including size information, from istream IS
66inline void read( stringVecT& stringVector, std::istream& iStream ){
67
68 if( iStream.peek() == EOF ){
69 perlish::die( __FILE__, __LINE__, "IO Error; string istream at EOF\n" );
70 }
71
72 // read size
73 size_t size;
74 iStream.read( (char*) &size, sizeof(size) );
75
76 stringVector.reserve( size );
77 for( size_t i = 0; i < size; ++i ){
78 std::string curString;
79 read( curString, iStream );
80 stringVector.push_back( curString );
81 }
82}

86// read size and then contents of string in from istream
87inline void read( std::string& s, std::istream& iStream ){
88
89 if( iStream.peek() == EOF ){
90 perlish::die( __FILE__, __LINE__, "IO Error; string istream at EOF\n" );
91 }
92
93 // read size
94 size_t size;
95 iStream.read( (char*) &size, sizeof(size) );
96
97 if( iStream.fail() ){
98 perlish::die( __FILE__, __LINE__, "IO Error; string istream read failed\n" );
99 }
100
101 // read string contents
102 read( s, iStream, size );
103}

107// read size elements from istream
108inline void read( std::string& s, std::istream& iStream, const size_t& size ){
109
110 // allocate space
111 if( size > s.max_size() ){
112 perlish::die( __FILE__, __LINE__, "IO Error; string size read in too large" );
113 }
114
115 s.reserve( size );
116 s.resize( size );
117
118 // read contents
119 iStream.read( (char*) &s[0], size*sizeof(s[0]) );
120
121 if( iStream.fail() ){
122 perlish::die( __FILE__, __LINE__, "IO Error; string istream read failed\n" );
123 }
124}

127// read size elements from istream
128inline void traceOrDie( const std::string& s, std::istream& iStream ){
129
130 // allocate space
131 std::string tmp;
132
133 tmp.resize( s.size() );
134
135 // read contents
136 iStream.read( (char*) &tmp[0], s.size() * sizeof( tmp[0] ) );
137
138 GDB_ASSERTF( !iStream.fail(),
139 "Input error, while trying to trace over string \"%s\"", s.c_str() );
140
141 GDB_ASSERTF( tmp == s,
142 "Input error, expected to trace over string \"%s\", but read \"%s\"",
143 s.c_str(), tmp.c_str() );
144}

147// write stringVector, including size information, to istream OS
148inline void write( const stringVecT& stringVector, std::ostream& os ){
149
150 // write number of strings
151 size_t size = stringVector.size();
152 os.write( (char*) &size, sizeof(size) );
153
154 for( size_t i = 0; i < stringVector.size(); ++i ){
155 write( stringVector[i], os );
156 }
157}
