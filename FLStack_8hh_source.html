30 void push( elemType b ) throw(std::length_error){
31 if( sizeVar == capacity() ){
32 char errorMessage[256];
33 snprintf( errorMessage, 256, " %s: line %d pushed element past stack capacity = %zu",
34 __FILE__, __LINE__, capacity() );
35#ifndef CBRC_QUIET_EXCEPTIONS

39 throw( std::length_error( errorMessage ) );
40 }
41 ar[sizeVar] = b;
42 ++sizeVar;
43 }
44 bool empty(){ return( sizeVar == 0 ); }
45 elemType& pop() throw(std::out_of_range){
46 if( empty() ) throw( std::out_of_range( " tried to pop empty stack" ) );
47 return ar[--sizeVar];
48 }
49 void zero(){ sizeVar = 0; }
50 elemType& operator[]( const size_t& index ) const throw(std::out_of_range){
51 if( index >= size() ){
52 char errorMessage[256];
53 snprintf( errorMessage, 256, "%s: line %d index = %zu not smaller than size = %zu",
54 __FILE__, __LINE__, index, size() );
55#ifndef CBRC_QUIET_EXCEPTIONS
56 std::cerr << errorMessage << std::endl;
57#endif // not defined CBRC_QUIET_EXCEPTIONS
58 throw( std::out_of_range( errorMessage ) );
59 }
60 return ar[index];
61 }
62 size_t capacity() const{ return ar.size(); }
63 size_t size() const{ return sizeVar; }
64
65 const_iterator begin() const{ return ar.begin(); }

68private:
