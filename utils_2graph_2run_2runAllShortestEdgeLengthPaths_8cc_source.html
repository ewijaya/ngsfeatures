15 void runAllShortestEdgeLengthPaths( bool summarizeOnly ){
16 NeighborListGraph nlg;
17 nlg.readOneEdgePerLineFormat( std::cin );
18 std::cerr << "num nodes read: " << nlg.numNodes() << std::endl;
19 // nlg.checkConsistency();
20 FLENumArray<nodeIndexT> histo( nlg.numNodes() );
21 histo.zero();
22 for( nodeIndexT i = 0; i < nlg.numNodes(); ++i ){
23 FLENumArray<nodeIndexT> edgeLen = nlg.shortestPathByEdgeLength( i );
24 FLEArray<size_t> perm = FLEArrayIdentityPermutation( edgeLen.size() );
25 std::stable_sort( perm.begin(), perm.end(), FLEArrayByIndexSortingPredicate<nodeIndexT>(edgeLen) );
26
27 if( !summarizeOnly ) std::cout << nlg.label(i);
28 for( nodeIndexT j = 0; j < perm.size(); ++j ){
29 nodeIndexT pathEnd = perm[j];
30 if( edgeLen[pathEnd] == nlg.numNodes() ) continue; // node j not reachable from i
31 if( pathEnd == i ) continue;
32 ++histo[ edgeLen[pathEnd] ];
33 if( summarizeOnly ) continue;
34 if( (j==0) || (edgeLen[pathEnd] != edgeLen[perm[j-1]]) ) std::cout << " " << edgeLen[pathEnd];
35 std::cout << " " << nlg.label(pathEnd);
36 }
37 if( !summarizeOnly ) std::cout << std::endl;
38 }
39 nodeIndexT graphDiameter = 0;
40 for( nodeIndexT i = 0; i < histo.size(); ++i ){
41 if( histo[i] ) graphDiameter = i;
42 }
43 std::cout << "graphDiameter: " << graphDiameter << std::endl;
44 if( graphDiameter == 0 ){
45 std::cout << "graph has no edges!\n";
46 return;
47 }
48 std::cout << histo[0];
49 for( nodeIndexT i = 1; i <= graphDiameter; ++i ){
50 std::cout << " " << histo[i];
51 }
52 std::cout << std::endl;
53 }
54
55}; // end namescape cbrc
