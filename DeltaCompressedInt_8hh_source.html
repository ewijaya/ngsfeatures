10 * Purpose: Compress a sorted list of ints into a more compact list using delta encoding.
11 *
12 * Caveat: Although defined as a template class this will only work as is with 4 byte ints (int & unsigned int)
13 *
14 * Data Representation: The first int is simply stored. For the following ints the difference is stored.
15 * i.e. for input v[0...], v[1]-v[0] is stored instead of explicitly storing v[1].
16 * The difference is stored as a variable length record similar to euc encoding.
17 * If the difference is < 2^7 one byte is used. If the difference is less than 2^14
18 * two bytes are used, etc. up to a maximum of five bytes.
19 *
20 * Usage: DelaCompressedInt<int> dc( std::vector<int> v ) // v must be sorted in ascending order
21 * int val = dc.first();
22 * for( size_t dcArrayIndex = 0; ; val += dc.nextDelta( dcArrayIndex ) ){
23 * if( dcArrayIndex > 0 ) os << " ";
24 * // Do something with val.
25 * // ...
26 * if( dcArrayIndex >= dc.arraySize() ) break; // Achtung! Do not forget this!
27 * }
28 *
29 *
30 */

39template <typename intType>
40class DeltaCompressedInt{
41public:
42 // input vector must be in ascending sorted order. Cannot call with empy vector.
43 DeltaCompressedInt<intType>( const std::vector<intType>& v ){
44 sizeVar = v.size();
45 arraySizeVar = computeArraySize( v );
46 a.setSize(arraySizeVar);
47 size_t s = 0;
48 firstVar = v[0];
49 for( size_t i = 1; i < sizeVar; ++i, ++s ){
50 unsigned int diff = v[i] - v[i-1];
51 a[s] = diff & 0x7F;
52
53 if( diff <= 0x7F ) continue;
54 a[s++] |= 0x80;
55 diff = diff >> 7;
56 a[s] = diff & 0x7F;
57
58 if( diff <= 0x7F ) continue;
59 a[s++] |= 0x80;
60 diff = diff >> 7;
61 a[s] = diff & 0x7F;
62
63 if( diff <= 0x7F ) continue;
64 a[s++] |= 0x80;
65 diff = diff >> 7;
66 a[s] = diff & 0x7F;
67
68 if( diff <= 0x7F ) continue;
69 a[s++] |= 0x80;
70 diff = diff >> 7;
71 a[s] = diff;
72 }
73 }
74 intType first() const{ return firstVar; }
75
76 // next modifies argument! call with 0 to get first delta. returns arraySize() if no more values are left.
77 intType nextDelta( size_t& arrayIndex ) const{
78 intType delta = a[arrayIndex] & 0x7F;
79 if( !(a[arrayIndex++] & 0x80) ) return delta;
80
81
82 delta |= ( (a[arrayIndex] & 0x7F) << 7 );
83 if( !(a[arrayIndex++] & 0x80) ) return delta;
84
85 delta |= ( (a[arrayIndex] & 0x7F) << 14 );
86 if( !(a[arrayIndex++] & 0x80) ) return delta;
87
88 delta |= ( (a[arrayIndex] & 0x7F) << 21 );
89 if( !(a[arrayIndex++] & 0x80) ) return delta;
90
91 delta |= ( (a[arrayIndex] & 0x7F) << 28 );
92 ++arrayIndex;
93 return delta;
94 }
95 bool operator==( const std::vector<intType>& v ){
96 if( v.size() != sizeVar ) return false;
97 intType val = first();
98 size_t vIndex = 0;
99 for( size_t dcArrayIndex = 0; ; val += nextDelta( dcArrayIndex ) ){
100 if( val != v[vIndex++] ) return false;
101 if( dcArrayIndex >= arraySize() ) break;
102 }
103 return true;
104 }
105
106 size_t size() const{ return sizeVar; }
107 size_t arraySize() const{ return arraySizeVar; }
108private:
109 size_t computeArraySize( const std::vector<intType>& v ){
110 assert( v.size() > 0 );
111 size_t s = 0;
112 for( size_t i = 1; i < v.size(); ++i ){
113 assert( v[i] > v[i-1] );
114 unsigned int diff = v[i] - v[i-1];
115 ++s;
116 if( diff > 0x7F ) ++s;
117 if( diff > 0x3FFF ) ++s;
118 if( diff > 0x1FFFFF ) ++s;
119 if( diff > 0xFFFFFFF ) ++s;
120 }
121 return s;
122 }
123 size_t sizeVar;
124 size_t arraySizeVar;
125 intType firstVar;
126 FLArray<unsigned char> a;
127};
