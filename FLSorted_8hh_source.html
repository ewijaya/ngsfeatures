33template <typename T>
34class FLSorted{
35public:
36 /* ********** CONSTRUCTORS ********** */
37 FLSorted(){}
38 FLSorted( const std::vector<T>& v ) : a(v){
39 std::sort( a.begin(), a.end() );
40 a.resize( unique( a.begin(),a.end() ) - a.begin() ); // remove duplicates and resize.
41 }
42
43 void operator=( const std::vector<T>& v ){
44
45 GDB_ASSERTF( a.empty(), "FLSorted object can only be assigned to once" );
46
47 a = v;
48 std::sort( a.begin(), a.end() );
49 a.resize( unique( a.begin(),a.end() ) - a.begin() );
50 }
51
52 void assign( const typename std::vector<T>::const_iterator& begin,
53 const typename std::vector<T>::const_iterator& end ){
54 if( !a.empty() ){
55 if( std::equal( begin, end, a.begin() ) ){
56 GDB_DIEF( "redundant assignment of FLSorted object" );
57 }else{
58 GDB_DIEF( "FLSorted object can only be assigned to once" );
59 }
60 a.assign( begin, end );
61 std::sort( a.begin(), a.end() );
62 a.resize( unique( a.begin(),a.end() ) - a.begin() );
63 }
64 }
65
66 friend std::ostream& operator<< <T>( std::ostream &os, const FLSorted<T>& sl );
67
68
69 /* ********** ACCESSORS ********** */
70 size_t size() const{ return a.size(); }
71 bool has( const T& t ) const{ return( std::binary_search( a.begin(), a.end(), t ) ); }
72
73 const T& operator()( const size_t& i ) const{ return a(i); }

83private:
