21void listConnectedComponents( std::string filename ){
22 std::ifstream is;
23 is.open( filename.c_str() );
24 nodeIndexT numNodes = OneEdgePerLineFormat::numNodesInStream( is ); // first count nodes.
25 is.clear(); is.seekg( 0, std::ios::beg ); // reset the filestream to be reread from the beginning.
26
27 ConnectedComponentOnlineComputer ccoc( numNodes );
28 SerialNumbersForStrings<nodeIndexT> labels;
29
30 // Read in graph and add edges.
31 INPUT_LINE:
32 while( !is.eof() ){
33 std::string line = perlish::slurpLine( is );
34 if( line.size() == 0 ) break;
35
36 std::vector<std::string> fields
37 = perlish::split( OneEdgePerLineFormat::inputDelimiter(), line );
38
39 if( !fields.size() || (fields.size() > 2) ){
40 std::cout << "Error: expected 1 or 2 fields per line: " << line << std::endl;
41 exit( -1 );
42 }
43
44 if( fields.size() == 1 ){
45 nodeIndexT n0 = labels.getSerialNumber( fields[0] );
46 ccoc.addNode( n0 );
47 goto INPUT_LINE;
48 }
49
50 nodeIndexT n0 = labels.getSerialNumber( fields[0] );
51 nodeIndexT n1 = labels.getSerialNumber( fields[1] );
52 if( fields[0] == fields[1] ){
53 // In order for labels to have the same size as numNodes it is important that this
54 // if statement comes *after* the node name is added to labels. The case in which this is important
55 // is a singleton node with a self-edge.
56 std::cerr << "Warning. Skipping self edge from line: " << line << std::endl;
57 goto INPUT_LINE;
58 }
59
60 ccoc.addEdge( n0, n1 );
61 }
62
63 // create permutation array which indirectly sorts the elements of nodeComponents
64 FLEArray<nodeIndexT> nodeComponents(numNodes);
65 nodeIndexT numComponents = ccoc.getNodeComponents( nodeComponents );
66 FLEArray<size_t> perm = FLEArrayIdentityPermutation( nodeComponents.size() );
67 std::stable_sort( perm.begin(), perm.end(), FLEArrayByIndexSortingPredicate<nodeIndexT>(nodeComponents) );
68
69 // print out the nodes sorted by component.
70 nodeIndexT curNodeIndex = 0;
71 for( nodeIndexT i = 0; i < numComponents; ++i ){
72 for( ; // no initialization
73 (curNodeIndex < nodeComponents.size()) && nodeComponents[perm[curNodeIndex]] == i;
74 ++curNodeIndex ){
75 std::cout << labels( perm[curNodeIndex] ) << "\t" << i << std::endl;
76 }
77 }
78}
79
80}; // end namespace cbrc
