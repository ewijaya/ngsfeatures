8 * Input: Two multifasta files
9 *
10 * Output: The closest match (by global affine gap measure) of each sequence in file1 to a sequence in file2
11 *
12 * Sketch: For each sequence in file1:
13 * A quick lower bound calculation is done for each sequence in file2
14 * The sequences in file2 are sorted by this bound in decreasing order.
15 * Exact calculation using lower bounds to improve running time is done for each sequence in
16 * file2.
17 *
18 * The reason for sorting by lower bound is to try to align close matches first. The scores
19 * obtained are used as lower bounds and it is good to get good lower bounds early
20 * for more aggressive pruning.
21 */

39 void allPairsNearestGlobalAffineScore(){
40
41 AminoScorePam120 aminoScore;
42
43
44 FLEArray<alignScoreT> seqs1Bounds( seqs1.size() );
45 FLEArray<size_t> seqs1byBound = FLEArrayIdentityPermutation( seqs1.size() );
46 if( seqs1.size() < 1 ){
47 std::cout << "Error: second sequence set is empty\n";
48 exit( -1 );
49 }
50
51 for( size_t i = 0; i < seqs0.size(); ++i ){
52 size_t j = 0;
53 while( ignoreSelf && ( seqs0[i].name() == seqs1[j].name() ) ){
54 seqs1Bounds[j] = alignScoreTMin;
55 ++j;
56 }
57 if( j >= seqs1.size() ) break;
58 // first get rough feasible bound for each sequence.
59 size_t bestMatchIndex = j;
60 SeqGlobalAffineAligner gas( aminoScore, seqs0[i].residueIndices() );
61 alignScoreT bestLowerBound = gas.feasibleBound( seqs1[j].residueIndices() );
62 seqs1Bounds[j] = bestLowerBound;
63 for( ; j < seqs1.size(); ++j ){
64 if( ignoreSelf && ( seqs0[i].name() == seqs1[j].name() ) ){
65 seqs1Bounds[j] = alignScoreTMin;
66 continue;
67 }
68 seqs1Bounds[j] = gas.feasibleBound( seqs1[j].residueIndices() );
69 if( seqs1Bounds[j] < bestLowerBound ){
70 bestLowerBound = seqs1Bounds[j];
71 bestMatchIndex = j;
72 }
73 }
74 // sort (indirectly) seqs1 by bound
75 std::stable_sort( seqs1byBound.begin(), seqs1byBound.end(),
76 FLEArrayByIndexSortingPredicate
77 <alignScoreT, std::greater<alignScoreT> >(seqs1Bounds)
78 );
79
80 // do the actual alignment calculations.
81 j = 0;
82 while( ignoreSelf && ( seqs0[i].name() == seqs1[seqs1byBound[j]].name() ) ) ++j;
83 if( j >= seqs1.size() ) break;
84 FLEArray<alignScoreT> suffixBounds = SeqGlobalAffineAligner::
85 bestPossibleSuffixScore( aminoScore, seqs1[seqs1byBound[j]].residueIndices() );
86 alignScoreT curBound = gas.scoreBounded( seqs1[seqs1byBound[j]].residueIndices(),
87 suffixBounds, bestLowerBound );
88 if( curBound > bestLowerBound ){
89 bestLowerBound = curBound;
90 bestMatchIndex = seqs1byBound[j];
91 }
92 for( ; j < seqs1.size(); ++j ){
93 if( ignoreSelf && ( seqs0[i].name() == seqs1[seqs1byBound[j]].name() ) ) continue;
94
95 FLEArray<alignScoreT> suffixBounds = SeqGlobalAffineAligner::
96 bestPossibleSuffixScore( aminoScore, seqs1[seqs1byBound[j]].residueIndices() );
97 curBound = gas.scoreBounded( seqs1[seqs1byBound[j]].residueIndices(),
98 suffixBounds, bestLowerBound );
99 if( curBound > bestLowerBound ){
100 bestLowerBound = curBound;
101 bestMatchIndex = seqs1byBound[j];
102 }
103 }
104 std::cout << seqs0[i].name() << " "
105 << seqs1[bestMatchIndex].name() << " "
106 << bestLowerBound << std::endl;
107 }
108 }
