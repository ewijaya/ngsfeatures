52 inline seqTypeT guessSeqType( const std::string& seq ){
53 if( aminoResidueIndexMap.validResidues( seq ) ) return amino;
54 if( DNAProperResidueIndexMap.validResidues( seq ) ) return DNA;
55 if( RNAProperResidueIndexMap.validResidues( seq ) ) return RNA;
56 return unknown;
57 }

63inline const ResidueIndexMap& residueIndexMapGuessedFromResidues( const std::string& residues ){
64 if( DNAProperResidueIndexMap.validResidues( residues ) ) return DNAProperResidueIndexMap;
65 if( DNAWithNResidueIndexMap.validResidues( residues ) ) return DNAWithNResidueIndexMap;
66 if( RNAProperResidueIndexMap.validResidues( residues ) ) return RNAProperResidueIndexMap;
67 if( aminoResidueIndexMap.validResidues( residues ) ) return aminoResidueIndexMap;
68 std::cout << "Error: failed to guess sequence type:\n";
69 exit( -1 );
70}

73inline const ResidueIndexMap& residueIndexMapFromName( const std::string& seqTypeName ){
74 if( seqTypeName == "DNA" ) return DNAProperResidueIndexMap;
75 if( seqTypeName == "dna" ) return DNAProperResidueIndexMap;
76 if( seqTypeName == "RNA" ) return RNAProperResidueIndexMap;
