36void clusterByIdentity(){
37
38 /* ** declare and read in sequences ** */
39 FastaSeqSlurper seqSlurper;
40
41 const std::vector<LabeledSequence>
42 seqs( seqSlurper.slurpLabeledSequences( *args.fastaIstreamPtr ) );
43
44 /* ** declare and initialized connected componentent computer ** */
45 ConnectedComponentOnlineComputer ccoc( seqs.size() );
46
47 for( size_t seqIdx = 0; seqIdx < seqs.size(); ++seqIdx ){
48 ccoc.addNode( seqIdx );
49 }
50
51
52 const AminoScoreIdentity aminoScoreIdentity( aminoResidueIndexMap );
53
54 for( size_t seqIdx0 = 0; seqIdx0 < seqs.size() - 1; ++seqIdx0 ){
55
56 const LabeledSequence& seq0 = seqs.at(seqIdx0);
57
58
59 SeqGlobalAffineAligner aligner( aminoScoreIdentity, seq0.residueIndices() );
60
61
62
63 for( size_t seqIdx1 = seqIdx0 + 1; seqIdx1 < seqs.size(); ++seqIdx1 ){
64
65 if( ccoc.getComponent( seqIdx0 ) == ccoc.getComponent( seqIdx1 ) ){
66 // seq0 and seq1 are known to be in the same cluster to skip alignment step
67 continue;
68 }
69
70
71 const LabeledSequence& seq1 = seqs.at(seqIdx1);
72
73 FLEArray<alignScoreT> suffixBounds =
74 SeqGlobalAffineAligner::bestPossibleSuffixScore( aminoScoreIdentity,
75 seq1.residueIndices() );
76
77 /* ** compute minimum alignment score for edge in cluster graph
78 * based on formula:
79 * adjusted identity = (# matches + adjust) / (max length + adjust)
80 */
81
82
83 const double maxLength = std::max( seq0.size(), seq1.size() );
84
85 const double& af = args.lengthAdjustmentFactor;
86
87 GDB_ASSERTF( af < maxLength,
88 "lengthAdjustmentFactor = %g, too big for averageLength = %g",
89 args.lengthAdjustmentFactor, maxLength );
90
91 double minScore
92 = args.identityThreshold * ( af + maxLength ) - af;
93
94 // compensate for fact that SeqGlobalAffineAligner currently penalizes terminal gaps
95 // would be better to modify SeqGlobalAffineAligner...
96 if( seq0.length() != seq1.length() ){
97 minScore += abs( seq0.length() - seq1.length() ) * aminoScoreIdentity.gapExtension();
98 }
99
100 const double boundedScore
101 = aligner.scoreBounded( seq1.residueIndices(),
102 suffixBounds,
103 (alignScoreT) minScore );
104
105 /* ** add edge if seq0 and seq1 are sufficiently similar ** */
106 if( boundedScore > minScore ){
107 ccoc.addEdge( seqIdx0, seqIdx1 );
108 }
109 } // end for seqIdx1
110 } // end for seqIdx0
111
112
113 /* ** extract connected component information and transform it
114 * ** so that the sequences can be printed with sequences in the
115 * ** same connected component grouped together
116 */
117 FLEArray<nodeIndexT> seqComponents( seqs.size() );
118 ccoc.getNodeComponents( seqComponents );
119
120 /* ** compute permutation of sequence indices PERM, such that
121 * ** i < j --> component[ perm[i] ] â‰¦ component[ perm[j] ]
122 */
123 const FLEArray<size_t> seqIdx_sortedByComponent
124 = FLEArrayIdentityPermutation( seqs.size() );
125
126 std::stable_sort( seqIdx_sortedByComponent.begin(),
127 seqIdx_sortedByComponent.end(),
128 FLEArrayByIndexSortingPredicate<nodeIndexT>(seqComponents) );
129
130
131 for( size_t seqIdx = 0; seqIdx < seqIdx_sortedByComponent.size(); ++seqIdx ){
132
133 /* ** skip singleton clusters ** */
134 const nodeIndexT curComponent = seqComponents(seqIdx);
135 if( ( (seqIdx == seqs.size()-1) || (seqComponents(seqIdx+1) != curComponent) )
136 && ( (seqIdx == 0) || (seqComponents(seqIdx-1) != curComponent) ) ){
137 continue;
138 }
139
140 // print cluster
141 std::cout << seqs.at( seqIdx ).name() << "\t"
142 << seqs.at( seqIdx ).label(1) << "\t"
143 << seqComponents( seqIdx ) << std::endl;
144
145 // mark cluster boundaries
146 if( (seqIdx < seqs.size()-1) && (seqComponents(seqIdx+1) != curComponent) ){
147 std::cout << "===\n";
148 }
149 }
150
151} // end function clusterByIdentity

157int main( int argc, const char* argv[] ){
158 cbrc::ArgvParser argP( argc, argv,
159 "-i identityThreshold [-a lengthAdjustmentFactor] [fastaFile]" );
160
161 argP.setDoc( "--man", "\
162OUTPUT\n\
163 sequenceID sequenceClassLabel clusterNumber\n\
164\n\
165OPTIONS\n\
166\n\
167 "Q(ADJUSTMENT_FLAGS)"\n\
168 Adjustment factor, positive value raises the \"identity score\" of long sequences\n\
169 relative to their actually identity.\n\
170\n\
171 adjusted identity = (# matches + adjust) / (max length + adjust)\n\
172");
173
174 argP.printDoc();
175
176 argP.setOrDie( args.identityThreshold, Q(IDENTITY_FLAGS) );
177
178 if( args.identityThreshold > 1.0 ){
179 argP.die( "identity Threshold should be in (0.0,1.0)" );
180 }
181
182 args.lengthAdjustmentFactor = 0.0; // default.
183 argP.set( args.lengthAdjustmentFactor, Q(ADJUSTMENT_FLAGS) );
184
185 args.fastaIstreamPtr = argP.getIstreamPtr( 1 );
186
187 argP.dieIfUnusedArgs();
188 cbrc::clusterByIdentity();
189 return 1;
