24class LabeledAsciiSeq{
25public:
26 /* ********** CONSTRUCTORS ********** */
27 LabeledAsciiSeq()
28 : _seq( stringUtils::null ),
29 _labels(0)
30 {}
31
32 LabeledAsciiSeq( const std::string& seq,
33 const std::vector<std::string>& labels ) :
34 _seq(seq){
35 std::copy( labels.begin(), labels.end(),
36 std::back_inserter( _labels ) );
37 }
38
39 LabeledAsciiSeq( const LabeledAsciiSeq& labeledAsciiSeq ) :
40 _seq( labeledAsciiSeq._seq )
41 {
42 _labels.reserve( labeledAsciiSeq.labels().size() );
43 std::copy( labeledAsciiSeq.labels().begin(),
44 labeledAsciiSeq.labels().end(),
45 std::back_inserter( _labels ) );
46 }
47
48 explicit LabeledAsciiSeq( const std::string& seq,
49 const std::string name = defaultSeqName() ) :
50 _seq(seq){
51 _labels.push_back( name );
52 }
53
54 // Assignment
55 const LabeledAsciiSeq& operator=( const LabeledAsciiSeq& labSeq ){
56 _seq = labSeq.seq();
57 _labels.clear();
58 _labels.reserve( labSeq.labels().size() );
59 std::copy( labSeq.labels().begin(),
60 labSeq.labels().end(),
61 std::back_inserter( _labels ) );
62 return *this;
63 }
64
65 /* ********** ACCESSORS ********** */
66 const std::string& seq() const { return _seq; }
67 /***/ void setSeq( const std::string& seq ) { _seq = seq; }
68
69 const std::string& name() const { return label(0); }
70 void setName( const std::string& name ) { _labels[0] = name; }
71
72 size_t size() const { return seq().size(); }
73
74 const char& operator()( const size_t& idx ) { return _seq[idx]; }
75
76
77 const std::string& label( const size_t& i ) const { return _labels.at(i); }
78 const std::vector<std::string>& labels() const { return _labels; }
79 const size_t numLabels() const { return _labels.size(); }
80
81 void setLabels( const std::vector<std::string>& labels ){
82 _labels = labels;
83 }
84
85private:
86
87 /* ** static methods ** */
88 static const std::string& defaultSeqName(){
89 static const std::string _defaultSeqName( "unknown" );
90 return _defaultSeqName;
91 }
92
93 /* ** object data ** */
94 std::string _seq;
95 std::vector<std::string> _labels;
96};
