109 for( size_t i = 0; i < v.size()-1; ++i ){
110 GDB_ASSERTF( v[i] < v[i+1],
111 "string number %zu = \"%s\", expected to be lexically less than"
112 "following string = \"%s\"",
113 i, v[i].c_str(), v[i+1].c_str() );
114 }
115 }
