19void NeighborListGraph::readOneEdgePerLineFormat( std::istream& is ){
20
21 // read in graph
22 while( !is.eof() ){
23 std::string line = perlish::slurpLine( is );
24 if( line.size() == 0 ) break;
25
26 std::vector<std::string> fields
27 = perlish::split( OneEdgePerLineFormat::inputDelimiter(), line );
28
29 GDB_ASSERTF( fields.size() == 2,
30 "Expected two fields when reading line: %s", line.c_str() );
31
32 if( fields[0] == fields[1] ){
33 std::cerr << "Warning. Skipping self edge from line: " << line << std::endl;
34 // okay to skip here because adding a singleton node does not change the desired output.
35 continue;
36 }
37
38 const nodeIndexT n0 = labels.getSerialNumber( fields[0] );
39 if( n0 >= ne.size() ) ne.push_back( std::vector<nodeIndexT>() );
40
41 const nodeIndexT n1 = labels.getSerialNumber( fields[1] );
42 if( n1 >= ne.size() ) ne.push_back( std::vector<nodeIndexT>() );
43
44 ne[n0].push_back( n1 );
45 ne[n1].push_back( n0 );
46 }
47
48 // sort neighbor lists.
49 for( size_t i = 0; i < ne.size(); ++i ){
50 std::sort( ne[i].begin(), ne[i].end() );
51 }
52}

80// Returns the number of edges on the shortest path from each node to n.
81// The number is set to numNodes() for nodes which are unreachable from n.
82FLENumArray<nodeIndexT>
83NeighborListGraph::shortestPathByEdgeLength( const nodeIndexT& n ) const{
84
85 assert( n < numNodes() );
86
87 FLENumArray<nodeIndexT> retVal( numNodes(), numNodes() );
88
89 BitArray reached( numNodes() );
90 reached.zero();
91 retVal[n] = 0;
92 reached.set( n );
93
94 FLStack<nodeIndexT> stack0( numNodes() );
95 FLStack<nodeIndexT> stack1( numNodes() );
96
97 std::vector< std::vector<nodeIndexT> > nodeStack( numNodes()-1 );
98 for( size_t i = 0; i < ne[n].size(); ++i ){
99 nodeIndexT curPathEnd = (ne[n])[i];
100 stack0.push( curPathEnd );
101 retVal[ curPathEnd ] = 1;
102 reached.set( curPathEnd );
103 }
104
105 bool done = true; // initialized to silence compiler warning.
106 for( size_t depth = 0; depth < nodeStack.size()-1; ++depth ){
107 done = true;
108 for( size_t i = 0; i < stack0.size(); ++i ){
109 nodeIndexT expandingNode = stack0[i];
110 for( size_t j = 0; j < ne[expandingNode].size(); ++j ){
111 nodeIndexT curPathEnd = (ne[expandingNode])[j];
112 if( !reached[ curPathEnd ] ){
113 stack1.push( curPathEnd );
114 retVal[curPathEnd] = depth+2;
115 reached.set( curPathEnd );
116 done = false;
117 }
118 }
119 }
120 if( done ) break;
121 done = true;
122 ++depth;
123 if( depth >= nodeStack.size() ) break;
124 stack0.zero();
125 for( size_t i = 0; i < stack1.size(); ++i ){
126 nodeIndexT expandingNode = stack1[i];
127 for( size_t j = 0; j < ne[expandingNode].size(); ++j ){
128 nodeIndexT curPathEnd = (ne[expandingNode])[j];
129 if( !reached[ curPathEnd ] ){
130 stack0.push( curPathEnd );
131 retVal[curPathEnd] = depth+2;
132 reached.set( curPathEnd );
133 done = false;
134 }
135 }
136 }
137 if( done ) break;
138 stack1.zero();
139 }
140 return retVal;
141}
