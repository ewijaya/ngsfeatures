12 * Purpose: Provide a general purpose hashtable which is easier to debug than sgi's hash_map. I do not claim
13 * that this code has any special merits other than that at this point. This code may not appear that easy to
14 * read, but try reading code from the STL...
15 *
16 * O'Reilly's Practical C++ Programming claims the way to make templates portable is to inline everything,
17 * and having everything in one file does have its advantages, thus this mammoth header file. Reading the skeleton
18 * API's in this section should make the rest easier to grok.
19 *
20 * ChainHashtable is a template class constructed with type hashableType. hashableType must supply a copy
21 * constructor, destructor, and two additional special methods: size_t hashVal() const;
22 * bool keyEquals( hashableType& ht ) const;
23 *
24 * Although hashVal() must be provided by the class of type hashableType, this file provides a utility function
25 * genericHashVal which often makes this method trivial to provide:
26 *
27 * size_t ChainHashtableUtils::genericHashVal( size_t numBytes, unsigned int* a);
28 *
29 * returns a general hash value for any contiguous data of length numBytes, starting at the place where
30 * a is pointing to. In general of course a cast will need to be used to make the pointer of type unsigned int*
31 * Note that the numBytes does not have to be a multiple of sizeof( unsigned int ).
32 *
33 * See the implementation below for details on the hash function.
34 *
35 *
36 * A full STL type iterator is not provided. Instead the following bare bones ChainHashIterator is provided:
37 *
38 * // skeleton of ChainHashIterator
39 * template<typename hashableType>
40 * class ChainHashIterator{
41 * public:
42 * hashableType* getElemPtr() const;
43 * bool operator==( ChainHashIterator& ch ) const;
44 * bool operator!=( ChainHashIterator& ch ) const;
45 * ChainHashIterator& ChainHashIterator::operator++();
46 * bool valid();
47 * }
48 *
49 * I think this iterator should be reasonably robust to concurrent additions to the table, but this is untested.
50 * The iterator is definitely not robust to having the table resized.
51 *
52 * The iterator is used in the following way:
53 *
54 * ChainHashtable<FreqMatrixChainHashable> table;
55 * for( ChainHashIterator<FreqMatrixChainHashable> it = table.begin(); it.valid(); ++it ){
56 * os << *(it.getElemPtr());
57 * }
58 *
59 * Here is a skeleton of the public methods ChainHashtable provides at this time:
60 *
61 * template<typename hashableType>
62 * // construct with size (actual size will be a prime number >= size), see code for default values.
63 * ChainHashtable<hashableType>( size_t size = default, maxTSize = default );
64 *
65 * // if element with key equal to elem is in the table return a pointer to it, otherwise return null.
66 * // This function is currently untested.
67 * hashableType* ChainHashtable::find( const hashableType& elem )
68 *
69 * // if elem is not found in table add it and return null.
70 * // else return a pointer to the element in the table which has the same key as elem (as per elem.keyEqual())
71 * // called with resizeAllowed set to true the table may get resized to accomodate the added element.
72 * // the condition for resizing is determined by "const unsigned int ChainHashtableMaxChainLen" in this method.
73 * hashableType* add( const hashableType& elem, bool resizeAllowed = true );
74 *
75 * // Clear the hash table of all elements, the hashableType destructor gets called for each cleared element.
76 * void ChainHashtable::clear()
77 *
78 * // self explanatory methods:
79 * size_t size() const{ return numElems; }
80 * ChainHashIterator<hashableType> begin() const;
81 *
82 * // for debugging.
83 * bool tableStateConsistent( size_t tableSize, ChainHashtableLink<hashableType>** tablePtr )
84 * *************************************************************************
85 *
86 * Note: At this point this class has been constructed for a very particular purpose and thus has a very
87 * limited set of methods. If this turns out to truly be a resuable class, methods such as delete will added.
88 */

141 static int dummy;
142 if( !elemPtr ) return false;

144 if( next ){
145 if( next->elemPtr ) ++dummy;

147 }

157// Santa's iterator elf.
158template<typename hashableType>
159class ChainHashIterator{
160public:
161 friend class ChainHashtable<hashableType>;
162 ChainHashIterator<hashableType>( ChainHashtableLink<hashableType>** head,
163 ChainHashtableLink<hashableType>** last ){
164 this->head = head; this->last = last;
165 if( head ) linkPtr = *head;
166 }
167 bool valid(){ return head; }
168 hashableType* getElemPtr() const{ return getLinkPtr()->getElemPtr(); }
169 bool operator==( ChainHashIterator& ch ) const{
170 if( head != ch.head ) return false;
171 return( linkPtr == ch.linkPtr );
172 }
173 bool operator!=( ChainHashIterator& ch ) const{ return( !(*this==ch) ); }
174 ChainHashIterator<hashableType>& operator++(){
175 if( !valid() ) return *this; // ++ing an invalid iterator is a no-op not an error.
176 if( linkPtr->getNext() ){ // walk down one step.
177 linkPtr = linkPtr->getNext();
178 return (*this);
179 }
180 if( head >= last ){
181 head = NULL;
182 return *this;
183 }
184 for( ++head; (head<last) && !*head; ++head );
185 if( *head ) linkPtr = *head;
186 else head = NULL;
187
188 return *this;
189 }
190private:
191 ChainHashtableLink<hashableType>* getLinkPtr() const{ return linkPtr; }
192 ChainHashtableLink<hashableType>** head;
193 ChainHashtableLink<hashableType>* linkPtr;
194 ChainHashtableLink<hashableType>** last;
195};

341 // utility which can make hashVal() trivial to write in many cases.
342 // It uses exclusive or and bit rotation to produce a hash function which should be robust to
343 // most cases of transposing elements or bytes (using only exclusive or is weak against transpositions)
344 inline size_t genericHashVal( const size_t& numBytes, const size_t* a){
345 size_t numWords = numBytes/4;
346 size_t numExtraBytes = numBytes - 4*numWords;
347 size_t key = 0;
348 const size_t* pastLastWord = a+numWords;
349 for( ; a < pastLastWord; ++a ){
350 key ^= *a;
351 BIT_ROTATE_LEFT( key );
352 }
353 const size_t b = * (size_t*) pastLastWord;
354#ifdef __x86_64__
355 static const size_t mask[] = {
356 0x0000000000000000,
357 0x00000000000000FF,
358 0x000000000000FFFF,
359 0x0000000000FFFFFF,
360 0x00000000FFFFFFFF,
361 0x000000FFFFFFFFFF,
362 0x0000FFFFFFFFFFFF,
363 0x00FFFFFFFFFFFFFF,
364 };
365 return( key ^ (b & mask[numExtraBytes]) );
366#elif __sparc__ // big endian
367 if( numExtraBytes == 1 ) return( key ^ (b & 0xFF000000) );
368 if( numExtraBytes == 2 ) return( key ^ (b & 0xFFFF0000) );
369 /* numExtraBytes == 3 */ return( key ^ (b & 0xFFFFFF00) );
370#elif __i386__
371 if( numExtraBytes == 1 ) return( key ^ (b & 0x000000FF) );
372 if( numExtraBytes == 2 ) return( key ^ (b & 0x0000FFFF) );
373 /* numExtraBytes == 3 */ return( key ^ (b & 0x00FFFFFF) );
374#else
375#error unknown architecture. only __x86_64__, __i386__ and __sparc__ currrently supported
376#endif // big or little endian
377 }
378}
