15// Set TransfacTextLine object by parsing a transfac style line string.
16// The string should have a two character label, optionally
17// followed by white space and some content.
18// If the line has not content, _content is set to the empy string.
19void TransfacTextLine::set( const std::string& line ){
20 std::vector<std::string> fields = perlish::split( line, perlish::spaceTabString(), 2 );
21 if( !fields.size() ){
22 std::cout << __FILE__ << ":" << __LINE__
23 << " Error: expected some text on transfac line.\n"
24 << "String recieved:\"" << line << "\".\n";
25 exit( -1 );
26 }
27 _label = fields[0];
28 if( _label.size() != 2 ){
29 std::cout << __FILE__ << ":" << __LINE__
30 << " Error: expected length 2 label string at start of tranfac line.\n"
31 << "String recieved:\"" << line << "\".\n";
32 exit( -1 );
33 }
34 _content = (fields.size() > 1 ) ? fields[1] : "";
35}
