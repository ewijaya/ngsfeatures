18bool FastafmtReader::readRecord( std::istream& is ){
19 bool retVal = readRecordAux( is );
20 if( retVal && seq().size() ){
21 std::string& seqRef = _curRecord.seqRef();
22 if( seqRef[ seq().size() - 1 ] == FastafmtConstants::seqTerminationMarker() ){

31bool FastafmtReader::readRecordAux( std::istream& is ){
32
33 _curRecord.setSeq( "" );
34
35 /* ******* set record head line ******* */
36 // if first record in stream, read in record head line
37 if( nextRecordHead.length() == 0 ){
38 if( is.eof() ) return false;
39 if( !getNoncommentLine( is, curRecordHead ) ) return false;
40 }
41 // else use line already read.
42 else{
43 curRecordHead = nextRecordHead;
44 }
45
46
47 { /* ******* Compute labelLine and labels from record head line ******* */
48 std::string labelLine = curRecordHead;
49
50 if( labelLine[0] != FastafmtConstants::recordMarker() ){
51 std::cerr << __FILE__ << ":" << __LINE__
52 << " Input Error: expected line to start with "
53 << FastafmtConstants::recordMarker() << std::endl
54 << "line: " << labelLine << std::endl;
55 return false;
56 }
57
58 std::string::const_iterator curPos = labelLine.begin() + 1;
59
60 assert(
61 advancePastMatchAtPoint( curPos,
62 labelLine.end(),
63 fastafmtInputOptions().beforeNameDelimiterRegex()
64 )
65 );
66
67 _curRecord.setLabels(
68 cbrc::perlish::split( fastafmtInputOptions().betweenLabelsDelimiterRegex(),
69 curPos,
70 std::string::const_iterator(labelLine.end()) )
71 );
72
73 } // end Compute labelLine and labels from record head line
74
75
76 /* ******* Read (one past) body of record ******* */
77 for( std::string curLine;
78 getNoncommentLine( is, curLine );
79 /* noop */ ){
80
81 // return if first line of next record is encountered
82 if( curLine[0] == '>' ){
83 nextRecordHead = curLine;
84 return true;
85 }
86
87 // erase spaces from line
88 curLine.erase(
89 std::remove( curLine.begin(), curLine.end(), ' ' ),
90 curLine.end()
91 );
92
93 _curRecord.seqRef() += curLine;
94 }
95
96
97 nextRecordHead = ""; // at end of file so there is no 1st line of next record.
98 return true;
99}

118// return true iff start of LINE matches commentMarker(), and commentMarker is not empty
119bool FastafmtReader::isComment( const std::string& line ) const{
120 if( commentMarker().empty() ) return false;
121 for( size_t i = 0;
122 (i < commentMarker().size()) && (i < line.size());
123 ++i ){
124 if( commentMarker()[i] != line[i] ) return false;
125 }
126 return true;
127}

130bool FastafmtReader::getNoncommentLine( std::istream& is, std::string& line ) const{
131 if( is.eof() ) return false;
132 while( std::getline( is, line ) ) {
133 if( !isComment(line) ) return true;
134 }
135 return false;
136}
