71// Optimized: pass by const reference, reserve space, use multiplication
72vector<double> normalizeP(const std::vector<double>& pn) {
73 double tot = 0.0;
74 for (unsigned i = 0; i < pn.size(); i++) {
75 tot += pn[i];
76 }
77
78 vector<double> nVec;
79 nVec.reserve(pn.size()); // Pre-allocate
80
81 const double inv_tot = 1.0 / tot; // Multiply is faster than divide
82 for (unsigned j = 0; j < pn.size(); j++) {
83 nVec.push_back(pn[j] * inv_tot);
84 }
85
86 return nVec;
87}

279vector<double> getPropSum(std::vector<string>& neigb, map<string, double>& m) {
280 vector<double> propVec;
281
282 // Find value of a neigbours
283 // from the map, otherwise return zero
284
285 for (unsigned i = 0; i < neigb.size(); i++) {
286 map<string, double>::iterator iter = m.find(neigb[i]);
287 double value = 0;
288
289 if (iter != m.end()) {
290 value = m[neigb[i]];
291 }
292 propVec.push_back(value);
293 // cout << neigb[i] << " " << value << endl;
294 }
295
296
297 // Sum the vector in three-three
298 // Resulting vector = nofpos
299
300 vector<double> pSum;
301 int nofPos = neigb.size() / 3;
302 int b = 0;
303 for (int k = 0; k < nofPos; k++) {
304 double output = propVec[b] + propVec[b + 1] + propVec[b + 2];
305 pSum.push_back(output);
306 b += 3;
307 }
308
309
310 return pSum;
311}
