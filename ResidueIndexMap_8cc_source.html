61// add residue to end of alphabet, see pod for more
62void ResidueIndexMap::extendAlphabet( const char& residue ) throw( std::invalid_argument ){
63 // if residue already in alphabet
64 // if it is the last residue do nothing, otherwise exit with error message
65 if( validResidue( residue ) ){
66 if( !residuesAreEquivalent( residue, alphabet()[alphabet().size()-1] ) ){
67 throw std::invalid_argument( "residue already present in alphabet" );
68 }
69 // else if residue is the last residue in the alphabet do nothing
70 // this makes repeated identical calls to extendAlphabet idempotent
71 return;
72 }
73 _alphabet.push_back( residue );
74}

124ResidueIndexMap::indexT
125ResidueIndexMap::toResidueIndex( const char& residue ) const throw( std::invalid_argument ){
126
127 for( indexT i = 0; i < alphabet().size(); ++i ){
128 if( residuesAreEquivalent( residue, alphabet()[i] ) ){

274// return true if residue0 == residue1 (possibly ignoring case)
275bool ResidueIndexMap::residuesAreEquivalent( const char& residue0, const char& residue1 ) const{
276 if( caseSensitive() ){
277 return( (residue0 == residue1) );
278 }
279
280 // else do case insensitive comparison
281 char r0 = tolower( residue0 );
282 char r1 = tolower( residue1 );
283 return( (r0 == r1) );
284}

290void ResidueIndexMap::read( std::istream& is ) throw( std::invalid_argument ){
291
292 // calling object should be empty
293 assert( !alphabet().size() );
294
295
296 /* ******* read in signature and confirm if it is correct ******* */
297
298 std::string signatureReadFromStream;
299
300 stringBinaryIO::read( signatureReadFromStream, is, binaryStreamSignature().size() );
301
302 if( signatureReadFromStream != binaryStreamSignature() ){
303 perlish::die( __FILE__, __LINE__, "Signature read from input stream does not match\n" );
304 }
305
306
307 /* ******* read in object data ******* */
308
309 // read in alphabetProper and alphabet
310 stringBinaryIO::read( _alphabetProper, is );
311 stringBinaryIO::read( _alphabet, is );
312
313 // read in caseSensitive
314 is.read( (char*) &_caseSensitive, sizeof(_caseSensitive) );
315 if( is.fail() ){
316 perlish::die( __FILE__, __LINE__, "Failed to read caseSensitive\n" );
317 }
318
319 // check validity
320 _assertAlphabetProperIsPrefix();
321 _assertAlphabetCharactersUnique();
322}

347std::vector<ResidueIndexMap::vectorT>
348ResidueIndexMap::splitIntoProperOnlySubstrings( const vectorT& residueIndexVector ) const{
349
350 std::vector<vectorT> retval;
351 size_t substringStart = 0;
352
353 while( true ){
354 // advance substringStart to next proper residue.
355
356 for( ; /* no initialization */
357 (substringStart < residueIndexVector.size())
358 && !properResidueIndex( residueIndexVector[substringStart] );
359 ++substringStart );
360
361 if( substringStart >= residueIndexVector.size() ) return retval;
362
363 // else residueIndexVector[substringStart] is a proper residue.
364 size_t substringEnd = substringStart + 1;
365 // advance substringEnd to next extra residue.
366
367 for( ; /* no initialization */
368 (substringEnd < residueIndexVector.size())
369 && properResidueIndex( residueIndexVector[substringEnd] );
370 ++substringEnd );
371
372 retval.push_back( vectorT(residueIndexVector.begin() + substringStart,
373 residueIndexVector.begin() + substringEnd) );
374 if( substringEnd >= residueIndexVector.size() ) return retval;
375 substringStart = substringEnd + 1;
376 }
377}

399// throw exception if the characters in alphabet are not unique
400// behavior depends on caseSensitive()
401void ResidueIndexMap::_assertAlphabetCharactersUnique() const throw( std::invalid_argument ){
402
403 std::string sortedAlphabet( alphabet() );
404
405 // sort the alphabet so that if duplicate characters are present,
406 // they will be adjacent to each other.
407 if( caseSensitive() ){
408 std::sort( sortedAlphabet.begin(), sortedAlphabet.end() );
409 }
410 else{
411 std::sort( sortedAlphabet.begin(), sortedAlphabet.end(), _caseInsensitiveSortPredicate );
412 }
413
414
415 // check for duplicate characters
416
417 for( size_t i = 1; i < sortedAlphabet.size(); ++i ){
418 if( residuesAreEquivalent( sortedAlphabet[i-1], sortedAlphabet[i] ) ){
419 throw std::invalid_argument( "characters in alphabet are not unique" );
420 }
421 }
422}

426bool ResidueIndexMap::_caseInsensitiveSortPredicate( const char& c0, const char& c1 ){
427
428 char lc0 = tolower( c0 );
429 char lc1 = tolower( c1 );
430
431 return( lc0 < lc1 );
432}
