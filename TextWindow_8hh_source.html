7 * Description: Represent a window into a "text". I use the word text here
8 * because this class was created for use with DNA sequences.
9 * Window here is synonymous with substring.
10 * However the template array T does not need to be a character
11 * type so in some instances the term "text" might be quite unnatural.
12 *
13 * Purpose: Originally created for background model used in motif
14 * discovery. But later a specialized class ResidueIndexWindow
15 * was made for that.
16 *
17 * Design Issues: This implementation stores four variables for
18 * each instance: start and end of both text and window.
19 * Another idea would be to just store one pointer to a
20 * text object that could return the start and end.
21 *
22 * Caveat: <T> should a fixed length type. Note that it might be okay for
23 * T to contain pointers to variable length data, for example std::string,
24 * If the size of the basic struct is fixed (I am not sure if std::string
25 * guarantees this or not...)
26 *
27 *
28 *
29 */

106 bool validRelativeIdx( int idx ){
107 if( idx < -static_cast<int>(windowStartIdx()) ){
108 std::cerr << __FILE__ << ":" << __LINE__ << "; Error: : idx = "
109 << idx << " less than -windowStartIdx() = "
110 << -static_cast<int>(windowStartIdx()) << std::endl;
111 return false;
112 }
113 if( idx >= static_cast<int>(textSize())
114 - static_cast<int>(windowStartIdx()) ){
115 std::cerr << __FILE__ << ":" << __LINE__ << "; Error: : idx ="
116 << idx << " greater than testSize()="
117 << static_cast<int>(textSize()) << " minus "
118 << " windowStartIdx()="
119 << static_cast<int>(windowStartIdx()) << std::endl;
120 return false;
121 }
122 return true;
123 }
124private:

129};
