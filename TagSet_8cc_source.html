36// returns: index of smallest sequence not lexically less than seqToMatch
37// or: size() if seqToMatch is lexically greater than any sequence tag.
38size_t TagSet::upperBound( const unpackedSeqT& seqToMatch ) const{
39
40 byte* const packedPtr
41 = new byte[sigma4bitPackingUtils::numBytesNeededToStore( seqToMatch.size() )];
42
43 sigma4bitPackingUtils::pack( packedPtr, seqToMatch );
44
45 size_t lower = 0;
46 size_t upper = size();
47
48 while( lower < upper ){
49 const size_t mid = lower + (upper - lower) / 2;
50 if( isLexicallyLess( mid, packedPtr, seqToMatch.size()) ) lower = mid + 1;
51 else upper = mid;
52 }
53
54 delete[] packedPtr;
55
56 return upper;
57}

64void TagSet::readFromTextStream( std::istream& iStream ){
65
66 std::string line;
67
68 DO_OR_DIEF( perlish::slurpLine( line, iStream ),
69 "Input error. Empty tag list stream?" );
70
71 const std::vector<std::string> fields = perlish::split( "[ \t]+", line );
72
73 ASSERTF( fields.size() == 2,
74 "Expected 2 fields in first line: \"%s\"", line.c_str() );
75
76 perlish::fromString( _size, fields[0] );
77
78 perlish::fromString( _totalSize, fields[1] );
79
80 ASSERTF( totalSize() >= size(),
81 "Total size (%zu) should be at least as big as size (%zu)", totalSize(), size() );
82
83
84 /* --------------- Allocate Memory --------------- */
85 a = new byte*[ size()+1 ];
86
87 ASSERTF( a,
88 "Failed to allocate array a of size (%zu)", size()+1 );
89
90 mem = new byte[ totalSize() ];
91
92 ASSERTF( mem,
93 "Failed to allocate array mem of size (%zu)", totalSize() );
94
95
96 /* ---------- Declare vars for read line loop ---------- */
97 ResidueIndexMap::vectorT residueIndices;
98
99 size_t tagCount = 0;
100 size_t totalResidueCount = 0;
101
102 /* --------------- Read line loop --------------- */
103 while( perlish::slurpLine( line, iStream ) ){
104 residueIndexMap().assignResidueIndices( residueIndices, line );
105
106 byte* const curMemStart = mem + totalResidueCount;
107
108 a[ tagCount ] = curMemStart;
109
110 sigma4bitPackingUtils::pack( curMemStart, residueIndices );
111
112 ++tagCount;
113 totalResidueCount += residueIndices.size();
114 }
115
116
117 /* ---------- Checksum Sequence and Residue Counts ---------- */
118 ASSERTF( tagCount == size(),
119 "Expected %zu tags, but only read in %zu", size(), tagCount );
120
121 ASSERTF( totalResidueCount == totalSize(),
122 "Expected %zu total redisidues, but only read in %zu",
123 totalSize(), totalResidueCount );
124
125
126 // added final value to a[] to allow computation of size of final sequence
127 a[ size() ] = mem + totalResidueCount;
128
129 assertIsSorted();
130
131} // end method readFromTextStream

152bool TagSet::isLexicallyLess( const byte* const& packedSeqPtr1,
153 const size_t& seqLen1, // len in characters
154 const byte* const& packedSeqPtr2,
155 const size_t& seqLen2 // len in characters
156 ) const{
157
158 const size_t minSizeInBytes
159 = std::min( sigma4bitPackingUtils::numBytesNeededToStore( seqLen1 ),
160 sigma4bitPackingUtils::numBytesNeededToStore( seqLen2 ) );
161
162 const int valCmp = memcmp( packedSeqPtr1, packedSeqPtr2, minSizeInBytes );
163
164 if( valCmp < 0 ) return true;
165 if( valCmp > 0 ) return false;
166
167 // sequences i, j are equal up to minSizeInBytes byte
168 return( seqLen1 < seqLen2 );
169}
