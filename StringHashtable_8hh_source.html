110// Auxillary class for StringHashtable
111// This iterator has three state variables. All pointers.
112// One is head. An iterator with a null head <==> an invalid iterator,
113// This is used for the end() of hashtables and the begin() of empty hashtables.
114class StringHashtableIterator{
115public:
116 StringHashtableIterator(){
117 invalidate(); // default constructor initializes to hashtable.end()
118 }
119 StringHashtableIterator( StringHashtableLink** head, StringHashtableLink** last ){
120 this->head = head; this->last = last;
121 if( head ) linkPtr = *head;
122 }
123 void invalidate(){ head = NULL; }
124 bool valid() const{ return head; }
125
126 static StringHashtableIterator makeInvalidIterator(){
127 return StringHashtableIterator();
128 }
129
130 bool operator==( const StringHashtableIterator& ch ) const{
131 if( head != ch.head ) return false;
132 if( !valid() ) return true; // only head needs to be compared for invalid iterators.
133 return( linkPtr == ch.linkPtr );
134 }
135
136 bool operator!=( const StringHashtableIterator& ch ) const{
137 return( !(*this==ch) );
138 }
139
140 KeyValuePair* elemPtr() const{ return getLinkPtr()->elemPtr(); }
141 KeyValuePair& operator*() const{ return *(elemPtr()); }
142
143 StringHashtableIterator& operator++(){
144
145 if( !valid() ){
146 return *this; // ++ing an invalid iterator is a no-op, not an error.
147 }
148
149 if( linkPtr->next() ){ // walk down one step.
150 linkPtr = linkPtr->next();
151 return *this;
152 }
153
154 if( head >= last ){
155 invalidate();
156 return *this;
157 }
158
159 for( ++head; (head<last) && !*head; ++head );
160 if( *head ) linkPtr = *head;
161 else invalidate();
162
163 return *this;
164 }
165
166 // object data.
167 StringHashtableLink* getLinkPtr() const{ return linkPtr; }
168 StringHashtableLink** head;
169 StringHashtableLink* linkPtr;
170 StringHashtableLink** last;
171};

198 void copy( const StringHashtable& ht ){
199 RUNTIME_WARN( "calling copy constructor for StringKeyhashtable\n" );
200 tsize = ht.tsize;
201 t = new StringHashtableLink*[tsize];
202 for( size_t i = 0; i < tsize; ++i ){
203 if( !ht.t[i] ){
204 t[i] = 0;
205 continue;
206 }
207
208 t[i] = new StringHashtableLink( ht.t[i]->elem().key(),
209 ht.t[i]->elem().value() );
210
211 StringHashtableLink* prevHT = ht.t[i];
212 StringHashtableLink* prev = t[i];
213 while( StringHashtableLink* curHT = prevHT->next() ){
214 prev->_next = new StringHashtableLink( curHT->elem() );
215 prev = prev->next();
216 prevHT = curHT;
217 }
218 }
219 }

307 std::string& insertKeyIfNew_returnValue( const std::string& key,
308 const bool resizeAllowed = true ){
309 const unsigned int StringHashtableMaxChainLen = 8;
310 const size_t index = stringHashValue( key ) % tsize;
311 unsigned int chainLen;
312 StringHashtableLink* linkPtr = t[index];
313
314 if( !linkPtr ){
316 t[index] = new StringHashtableLink( key, defaultInsertionValue() );
317 ++numElems;
318 return( t[index]->valueRef() );
319 }
320
321 // head of list was not null, so walk down linked list
322 for( chainLen = 2;
323 !(key==linkPtr->elem().key()) && linkPtr->next();
324 linkPtr=linkPtr->next(), chainLen++
325 );
326
327 if( key == linkPtr->elem().key() ){
328 return linkPtr->valueRef(); // EXIT. element already in table
329 }
330 // need to allocate new slot, but first check if we need to resize.
331 if( resizeAllowed
332 && (tsize < maxAllowedLinkListLength())
333 && (chainLen > StringHashtableMaxChainLen) ){
334 resizeLinkedList( tsize+1 );
335 return insertKeyIfNew_returnValue( key, false );
336 }
337 ++numElems;
338 linkPtr->setNext( new StringHashtableLink( key, defaultInsertionValue() ) );
339 return linkPtr->valueRef();
340 } // end insertKeyIfNew_returnValue

342 KeyValuePair* add( const KeyValuePair& elem, const bool resizeAllowed = true ){
343 const unsigned int StringHashtableMaxChainLen = 8;
344 const size_t index = elem.hashValue() % tsize;
345 unsigned int chainLen;
346 StringHashtableLink* linkPtr = t[index];
347
348 if( !linkPtr ){
350 t[index] = new StringHashtableLink(elem);
351 ++numElems;
352 return NULL; // elem was not found in the table so return null as per the API.
353 }
354
355 // head of list was not null so walk down\n";
356 for( chainLen = 2;
357 !elem.keyEquals( linkPtr->elem() ) && linkPtr->next();
358 linkPtr=linkPtr->next(), chainLen++ );
359
360 if( elem.keyEquals( linkPtr->elem()) ){
361 // std::cout << "element already in table\n";
362 return linkPtr->elemPtr();
363 }
364 // need to allocate new slot, but first check if we need to resize.
365 if( resizeAllowed
366 && (tsize < maxAllowedLinkListLength())
367 && (chainLen > StringHashtableMaxChainLen) ){
368 resizeLinkedList( tsize+1 );
369 return add( elem, false );
370 }
371 ++numElems;
372 linkPtr->setNext( new StringHashtableLink(elem) );
373 return NULL; // elem was not found in the table so return null as per the API.

411/* ********** METHOD IMPLEMENTATIONS ********** */
412inline void StringHashtable::resizeLinkedList( const size_t& minNewSize ){
413 const size_t newSize = nextPrime( minNewSize );
414 StringHashtableLink** newT = new StringHashtableLink*[newSize];
415 memset( newT, 0, sizeof(StringHashtableLink*)*newSize );
416 StringHashtableIterator curIt = begin();
417 StringHashtableIterator nextIt = curIt;
418 GDB_ASSERTF( curIt.valid() );
419 GDB_ASSERTF( nextIt.valid() );
420 for( ++nextIt;
421 curIt.valid();
422 curIt = nextIt, ++nextIt ){ // curIt state gets modified so can't ++curIt.
423 addNewToTablePtr( curIt.getLinkPtr(), newT, newSize );
424 }
425 StringHashtableLink** temp = t;
426 t = newT;
427 tsize = newSize;
428 delete temp;
429}
