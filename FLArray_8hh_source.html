8 * Purpose: Provide a fixed length array class that checks for the validity of indices.
9 *
10 * Description: FLArray is a fixed length array that checks the unsigned int index for being
11 * too large when using the operator [].
12 *
13 * FLArrayFast is a derived class that avoids that small overhead by not checking
14 * the index passed to operator [] for being too large. However, when debugging
15 * FLARRAY_ALWAYS_CHECK_INDICES can be defined to force index checking with
16 * FLArrayFast too.
17 *
18 * FLNumArray is an extension of FLArray which assumes T can do arithmetic.
19 * It supplies methods such as sum(), which returns the summation of the elements.
20 *
21 * FLNumArrayFast is exactly analogous to FLArrayFast.
22 * In fact I tried to have it multiple inherit from FLNumArray and FLArrayFast
23 * -- but I couldn't get that to work...
24 */

235 T sum() const{
236 T retVal = 0;
237 for( size_t i = 0; i < this->size(); ++i ) retVal += this->a[i];
238 return retVal;
239 }
240 T max() const{
241 assert( this->size() > 0 );
242 T retVal = this->a[0];
243 for( size_t i = 1; i < this->size(); ++i ) if( retVal < this->a[i] ) retVal = this->a[i];
244 return retVal;
245 }
246 size_t maxIndex() const{ // returns first index in case of ties.
247#ifdef GDB_DEBUG
248 if( FLArray<T>::size() == 0 ){
249 std::cout << __FILE__ << ":" << __LINE__ << " Error size() == 0!\n In gdb hit cntl-c...\n";
250 for(;;);
251 }
252#endif // defined GDB_DEBUG
253 assert( this->size() > 0 );
254 size_t retIndex = 0;
255 T maxVal = this->a[0];
256 for( size_t i = 1; i < this->size(); ++i ){
257 if( maxVal < this->a[i] ){ maxVal = this->a[i]; retIndex = i; }
258 }
259 return retIndex;
260 }
261 void squareElements(){
262 for( size_t i = 0; i < this->size(); i++ ) this->a[i] = this->a[i] * this->a[i];
263 }
