28 void labeledBlastToOddsRatio( std::istream& trainStream, std::istream& blastResultsStream ){
29 Blast8ResultReader blastResults( blastResultsStream, true );
30 LabeledInstances lis( LabeledInstances::text, trainStream );
31 BlastEValuesByClass blastClassifier( blastResults, lis );
32
33 std::vector<smoothing::NPOINT> correctByEValue;
34 double log2r = 1.0 / log( 2.0 );
35
36 FLEArray<double> eValues( lis.size() );
37 for( size_t i = 0; i < lis.size(); ++i ){
38 if( !blastResults.has( lis(i).name() ) ) continue;
39 std::vector< std::pair<std::string, double> > hitList = blastClassifier.bestEValuePerClass( lis(i).name() );
40 eValues[i] = hitList[0].second;
41 }
42 FLEArray<size_t> eValueSortedIndices = FLEArrayIdentityPermutation( eValues.size() );
43 std::stable_sort( eValueSortedIndices.begin(), eValueSortedIndices.end(), FLEArrayByIndexSortingPredicate<double>(eValues) );
44 for( size_t index = 0; index < lis.size(); ++index ){
45 size_t i = eValueSortedIndices[index];
46 if( !blastResults.has( lis(i).name() ) ) continue;
47 std::vector< std::pair<std::string, double> > hitList = blastClassifier.bestEValuePerClass( lis(i).name() );
48 std::string className = lis.classIdxToName( lis(i).classIdx() );
49 double correct = ( hitList[0].first == className ) ? 1.0 : 0.0;
50 // correctByEValue.push_back( WeightedPoint2D<double, double>( log2r * log(hitList[0].second), correct ) );
51 correctByEValue.push_back( smoothing::NPOINT( log2r * log(hitList[0].second), correct ) );
52 }
53 bool minValid = false;
54 double minVal = 0; // silence compiler.
55
56 // find smallest value not equal to -inf.
57 std::cout << __FILE__ << ":" << __LINE__ << "correctByEValue.size(): " << correctByEValue.size() << std::endl;
58 double log0 = log2r * log(0.0);
59 for( size_t j = 0; j < correctByEValue.size(); ++j ){
60 std::cout << __FILE__ << ":" << __LINE__ << " j: " << j << " ";
61 std::cout << "correctByEValue[j].x(): " << correctByEValue[j].x() << std::endl;
62 // if( correctByEValue[j].x() <= -std::numeric_limits<double>::infinity() ) continue;
63 // if( moreNumericLimits::isNegativeInfinity( correctByEValue[j].x() ) ) continue;
64 if( correctByEValue[j].x() == log0 ) continue;
65 if( minValid ) if( correctByEValue[j].x() < minVal ) minVal = correctByEValue[j].x();
66 else{
67 std::cout << __FILE__ << ":" << __LINE__ << "j: " << j << " ";
68 std::cout << "correctByEValue[j].x(): " << correctByEValue[j].x() << std::endl;
69 minValid = true;
70 minVal = correctByEValue[j].x();
71 }
72 }
73 std::cout << __FILE__ << ":" << __LINE__ << "minVal: " << minVal << std::endl;
74 assert( minValid );
75 // replace -inf with smallest finite value
76 for( size_t j = 0; j < correctByEValue.size(); ++j ){
77 // if( correctByEValue[j].x() == -std::numeric_limits<double>::infinity() ) correctByEValue[j].setX( minVal );
78 // if( moreNumericLimits::isNegativeInfinity( correctByEValue[j].x() ) ){
79 if( correctByEValue[j].x() == log0 ){
80 std::cout << __FILE__ << ":" << __LINE__ << " setting " << j << " to: " << minVal;
81 std::cout << " current val: " << correctByEValue[j] << std::endl;
82 correctByEValue[j].setX( minVal );
83 }
84 }
85
86 correctByEValue = smoothing::mergeIdenticalAdjacentX( correctByEValue );
87 std::vector<smoothing::POINT> smoothedByEValue = smoothing::monotonicSmoothY( correctByEValue, false );
88 std::cout << "correct by eValue:\n";
89 for( size_t i = 0; i < correctByEValue.size(); ++i ){
90 std::cout << correctByEValue[i].weight() << ": " << correctByEValue[i].x() << " " << correctByEValue[i].y();
91 std::cout << " | " << smoothedByEValue[i].x() << " " << smoothedByEValue[i].y() << std::endl;
92 }
93 std::cout << std::endl;
94 }
95}; // end namescape cbrc
