19ResidueIndexMapTranslator
20::ResidueIndexMapTranslator( const ResidueIndexMap& nucleicResidueIndexMap,
21 const ResidueIndexMap& aminoResidueIndexMap,
22 const std::string& codonTableDescriptor )
23 : nucleicResidueIndexMap( nucleicResidueIndexMap ),
24 aminoResidueIndexMap( aminoResidueIndexMap ),
25 codonTableDescriptor( codonTableDescriptor )
26{
27
28 /* ** initialize text parsing constants ** */
29 const std::string interCodonDelimiter( "\t+" );
30
31 const std::string intraCodonDelimiter( ":" );
32
33 const std::string stopCodonLiteral( "stop" );
34
35
36 // initialize toAminotide
37 toAminotide.setSize( numPossibleCodonIndices_upperBound() );
38 toAminotide.fill( UNDERTERMINED_CODON_INDEX );
39
40
41 stringVecT codonDescriptors
42 = perlish::split( interCodonDelimiter, codonTableDescriptor );
43
44 for( stringVecT::const_iterator curCodonPtr = codonDescriptors.begin();
45 curCodonPtr != codonDescriptors.end();
46 ++curCodonPtr ){
47
48 /* ** separate strings describing codon and amino acid ** */
49 stringVecT codonAminoPair
50 = perlish::split( intraCodonDelimiter, *curCodonPtr );
51
52 assert( codonAminoPair.size() == 2 ); // codon & amino acid
53
54 const std::string& codonString = codonAminoPair[0];
55 const std::string& aminoResidueString = codonAminoPair[1];
56
57
58 // residue indexed version of current codon
59 const ResidueIndexMap::arrayT residueIndexedCodon
60 = nucleicResidueIndexMap.toResidueIndices( codonString );
61
62 // index into toAminotide for current codon
63 const size_t curCodonIndex = codonIndex( &residueIndexedCodon[0] );
64
65
66 if( aminoResidueString == stopCodonLiteral ){
67
68 toAminotide[curCodonIndex] = STOP_CODON_INDEX;
69
70 }
71 else{
72 /* ** If codon for ordinary amino acid residue ** */
73
74 GDB_ASSERTF( aminoResidueString.size() == 1,
75 "expected single character representing an amino acid,"
76 "but instead got string \"%s\"", aminoResidueString.c_str()
77 );
78
79 const ResidueIndexMap::residueT aminoResidue = aminoResidueString[0];
80
81 GDB_ASSERTF( aminoResidueIndexMap.validResidue( aminoResidue ),
82 "expected single character representing an amino acid,"
83 "but instead got character (%c), with ascii code (%u)",
84 aminoResidue, (unsigned) aminoResidue
85 );
86
87 // set appropriate entry of toAminotide, to map codon to its amino acid residueIndex
88 toAminotide[curCodonIndex] = aminoResidueIndexMap.toResidueIndex( aminoResidue );
89
90 } // end stop/ordinary codon if
91
92 } // end for current codon descriptor
93
94} // end constructor ResidueIndexMapTranslator( nucleicResidueIndexMap,, )

98void ResidueIndexMapTranslator
99::translateInPlace( const residueIndexT* const& nucleicBeg,
100 const residueIndexT* const& nucleicEnd,
101 /***/ residueIndexT* const& aminoBeg,
102 const residueIndexT*& /***/ aminoEnd ) const{
103
104 assert( nucleicEnd >= nucleicBeg );
105
106 /* ** Initailize pointers ** */
107 residueIndexT* curAminoPtr = aminoBeg;
108
109 const residueIndexT* finalCodonEnd
110 = nucleicEnd - ( (nucleicEnd - nucleicBeg) % 3 );
111
112 const residueIndexT* curNucleotidePtr = nucleicBeg;
113
114
115 /* ** Loop, processing one codon at a time ** */
116 while( curNucleotidePtr < finalCodonEnd ){
117
118 // LOOP PROGRESS happens in nextCodonIndex.
119 const size_t curCodonIndex = nextCodonIndex( curNucleotidePtr );
120
121 GDB_ASSERTF( toAminotide( curCodonIndex ) != UNDERTERMINED_CODON_INDEX,
122 "encountered undetermined codon, corresponding bytes are: %u:%u:%u",
123 (unsigned) *curNucleotidePtr,
124 (unsigned) *(1+curNucleotidePtr),
125 (unsigned) *(2+curNucleotidePtr)
126 );
127
128 if( toAminotide( curCodonIndex ) == STOP_CODON_INDEX ){
129 break; // encountered stop codon, so EXIT.
130 }
131
132 // else translate codon and advance curAminoPtr.
133 *curAminoPtr++ = toAminotide( curCodonIndex );
134
135 } // end open reading frame while loop
136
137 aminoEnd = curAminoPtr; // "return" value of aminoEnd to caller.
138
139} // end method translateInPlace.

143size_t ResidueIndexMapTranslator::numPossibleCodonIndices_upperBound() const{
144
145 size_t retVal = nucleicResidueIndexMap.sigma()
146 * nucleicResidueIndexMap.sigma()
147 * nucleicResidueIndexMap.sigma();
148
149 GDB_ASSERTF( retVal < STOP_CODON_INDEX,
150 "σ^3 (%zu) ≧ %u, exceeding the range of ResidueIndexMap::indexT",
151 retVal, STOP_CODON_INDEX
152 );
153
154 return retVal;
155} // end method numPossibleCodonIndices

160::codonIndex( const residueIndexT* const nucleotidePtr ) const{
161
162 const size_t sigma = nucleicResidueIndexMap.sigma();
163 /***/ size_t codonIndex;
164
165 codonIndex = sigma * *nucleotidePtr;
166 codonIndex += *(1+nucleotidePtr);
167 codonIndex *= sigma;
168 codonIndex += *(2+nucleotidePtr);
169
