47alignScoreT Alignment::score( const AminoScore& ad ) const{
48 alignScoreT retVal = 0;
49 bool inGap = false;
50 for( size_t i = 0; i < size(); ++i ){
51 if( (s0[i] == indelChar()) || (s1[i] == indelChar()) ){
52 if( !inGap ){
53 inGap = true;
54 retVal += ad.gapInitiation();
55 }
56 retVal += ad.gapExtension();
57 continue;
58 }
59 inGap = false;
60 retVal += ad.score( s0[i], s1[i] );
61 }
62 return retVal;
63}

65double Alignment::percentIdentity() const{
66 size_t numMatches = 0;
67 for( size_t i = 0; i < size(); ++i ) numMatches += s0[i] == s1[i];
68 return( 200 * numMatches / static_cast<double>( s0.size() + s1.size() ) );

207void Alignment::outputTextMarkSame( std::ostream& os,
208 const bool markDifferent, const char markChar ) const{
209 assert( s0.size() == s1.size() );
210 const unsigned int rowLength=75;
211 size_t numBasesSoFar0 = 0; // number of bases from sequence 0 printed so far.
212 size_t numBasesSoFar1 = 0;
213 size_t maxDigits = 1;
214 for( size_t i = 10; s0.size() >= i; ++maxDigits, i *= 10 );
215 std::string numBasesSoFarFormat( std::string( "%" )
216 + perlish::toString( maxDigits )
217 + std::string( "d" ) );
218
219 for( unsigned int i = 0; i < size(); /* i is incremented inside this loop */ ){
220 if( i ) std::cout << std::endl;
221
222 // print this row's portion of s0
223 for( unsigned int j = 0; j < rowLength; ++j ){
224 unsigned int idx = i+j;
225 if( idx >= size() ) break;
226 if( s0[idx] == indelChar() ) os << indelChar();
227 else{
228 if( (size_t) s0[idx] > rim.sigma() ){ // invalid base character, print as number.
229 os << "(" << (size_t) s0[idx] << ")";
230 }else{
231 os << rim.toResidue( s0[idx] );
232 }
233 ++numBasesSoFar0;
234 }
235 }
236 os << " " << perlish::toString( numBasesSoFar0, numBasesSoFarFormat ) << std::endl;
237
238 // print this row's portion of s1
239 for( unsigned int j = 0; j < rowLength; ++j ){
240 unsigned int idx = i+j;
241 if( idx >= size() ) break;
242 if( s1[idx] == indelChar() ) os << indelChar();
243 else{
244 os << rim.toResidue( s1[idx] );
245 ++numBasesSoFar1;
246 }
247 }
248 os << " " << perlish::toString( numBasesSoFar1, numBasesSoFarFormat ) << std::endl;
249
250 // print same or different mark.
251 // Actung! In this (and only this loop) i gets incremented.
252 for( unsigned int j = 0; j < rowLength; ++j, ++i ){
253 unsigned int idx = i;
254 if( idx >= size() ) break;
255 if( s1[idx] == indelChar() ) os << " ";
256 else{
257 if( (s0[idx] == s1[idx]) != markDifferent ) os << markChar;
258 else os << " ";
259 }
260 }
261 os << std::endl;
262 }
263}
