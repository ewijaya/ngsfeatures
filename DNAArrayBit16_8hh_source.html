8 * Purpose: pack up to 8 DNA bases in a single byte. Mapping characters to binary 00, 01, 10, 11.
9 *
10 * Many logical error are checked for.
11 *
12 * Trying to construct a DNAArrayBit16 with a length greater than maxLength (= 8 bases)
13 * throws a std::length_error.
14 *
15 * Trying to access an element beyond the length of the array throws a
16 * std::out_of_range error. Giving an integer greater than maxResidueIndex (=3) for
17 * a base or an integer greater than 4^len-1 for a string of len bases
18 * (e.g. in setBases()) throws a std::invalid_argument error. Unless silenced by
19 * defining CBRC_QUIET_EXCEPTIONS, errors are printed to std:cerr before throwing.
20 *
21 * Example Usage: DAArrayBit16<6> shortArray( DNAResidueIndexMap.getResidueIndices( "gactta" ) );
22 * shortArray.substitute( 1, DNAResidueIndexMap.getResidue( 't' );
23 * std::cout << shortArray; // prints "gtctta"
24 *
25 * See also: ResidueIndexMap
26 */

140 void set( const size_t index, const byte base ) throw( std::out_of_range, std::invalid_argument ){
141 checkIndex( index );
142 checkBase( base );
143 DNAArrayBit16_bases8 b = base;
144 b <<= index; b <<= index;
145 bases &= DNAArrayBit16_clearMask[index];
146 bases |= b;
147 }
