29std::vector<size_t>
30SimilaritySearchableSequences::withinHammingDistance( const LabeledSequence& query,
31 const double& threshold ) const{
32 std::vector<size_t> retVal;
33
34 for( size_t curSeqIdx = 0; curSeqIdx < seqs.size(); ++curSeqIdx ){
35 if( HammingDistanceComputer::computeBounded( seqs[curSeqIdx], query, threshold )
36 < threshold ){
37 // num mismatches < threshold so push current sequence onto return list
38 retVal.push_back( curSeqIdx );
39 }
40 }
41
42 return retVal;
43}

65SimilaritySearchableSequences::indexScorePairT
66SimilaritySearchableSequences::
67bestMatch( const LabeledSequence& query,
68 const alignScoreT loosenessFactor,
69 const LabeledSequencePairPredicate& skipPairPredicate
70 ) const{
71
72 GDB_ASSERTF( seqs.size(), "Attempted to search against empty sequence set" );
73
74 GDB_ASSERTF( !wouldSkipAll( query, skipPairPredicate ),
75 "For query %s, all sequence pairs would be skipped",
76 query.name().c_str() );
77
78
79 const SeqGlobalAffineAligner gas( resSubScore(), query.residueIndices() );
80
81
82 /* ***** Calculate quick lower bounds and compute order in which to compare seqs ***** */
83
84
85 const alignScoreT lowestAcceptableScore =
86 loosenessFactor
87 ? gas.bestPossibleScore() - loosenessFactor * static_cast<alignScoreT>( sqrt( query.length()) )
88 : std::numeric_limits<alignScoreT>::min();
89
90
91 alignScoreT lowerBoundToUse = lowestAcceptableScore;
92
93 // first get rough feasible bound for each sequence.
94 alignScoreT bestLowerBound = std::numeric_limits<alignScoreT>::min();
95
96 // Compute lower bounds for each sequence aligned to anything.
97 for( size_t i = 0; i < seqs.size(); ++i ){
98
99 seqBounds[i] = gas.feasibleBound( seqs[i].residueIndices() );
100
101 bestLowerBound = std::min( bestLowerBound, seqBounds[i] );
102
103 }
104
105 if( lowerBoundToUse < bestLowerBound ){
106 lowerBoundToUse = bestLowerBound;
107 }
108
109
110 /* sort (indirectly) seqs by bound, this heuristically improves the chance of finding a match
111 * early in the list.
112 */
113 const FLEArray<size_t> seqsByBound = FLEArrayIdentityPermutation( seqs.size() );
114
115 std::stable_sort( seqsByBound.begin(), seqsByBound.end(),
116 FLEArrayByIndexSortingPredicate< alignScoreT, std::greater_equal<alignScoreT> >(seqBounds) );
117
118
119 /* ***** Do the actual alignment calculations ***** */
120
121 size_t bestMatchIdx = seqs.size(); // initialize to invalid value.
122 for( size_t permutationIndex = 0;
123 permutationIndex < seqsByBound.size();
124 ++permutationIndex ){
125
126 const size_t seqIdx = seqsByBound[permutationIndex];
127
128 if( skipPairPredicate( seqs[seqIdx], query ) ){
129 continue; // skip this pair of sequences
130 }
131
132 const alignScoreT scoreApprox
133 = gas.scoreSuffixBounded( seqs[seqIdx].residueIndices(), bestLowerBound );
134
135 if( scoreApprox > bestLowerBound ){ // in this case scoreApprox = real score
136 bestLowerBound = scoreApprox;
137 bestMatchIdx = seqIdx;
138 }
139 } // next permutationIndex
140
141 GDB_ASSERTF( ( bestMatchIdx < seqs.size() ),
142 "bestMatchIdx value (%zu) invalid. Perhaps never set?",
143 bestMatchIdx );
144
145 indexScorePairT retVal;
146
147 if( bestLowerBound >= lowestAcceptableScore ){ // bestLower is exact.
148 retVal.first = bestMatchIdx;
149 retVal.second = bestLowerBound;
150 }else{
151 // bestLower is not exact and may not even be a valid lower bound.
152 retVal.first = seqs.size(); // indicates invalid match
153 }
154
155 return retVal;
156
157} // end SimilaritySearchableSequences::bestMatch( const LabeledSequence& query,...
