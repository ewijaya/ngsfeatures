12 * Purpose: Provide a fixed matrix (2D array) class that checks for the validity of indices.
13 *
14 *
15 * Description: FLEMatrix is a fixed length 2D array that can also be efficiently accessed as a 1D array. *
16 * Just as FLEArray and FLArray, FLEMatrix was derived from FLMatrix.
17 * The FLEMatrix family of classes can correctly hold fixed length elements. It
18 * is *not* correct for variable length type elements or pointer containing types
19 * such as std::string. It also does not guarantee calling constructors or
20 * destructors on the contained elements. The FLMatrix family of classes is provided
21 * for variable length types such as std::string
22 *
23 * The operator () is used to access as a 2D array.
24 * The operator [] is used to access as a 1D array.
25 * e.g. FLEMatrix<double> fml( 2, 3 );
26 * fml( 1, 2 ) = 3.2;
27 * fml[ 2*1+2 ] = 3.2; // these two lines are equivalent.
28 *
29 * Another way to access elements is to get a column pointer with getColPtr( size_t index0 ).
30 *
31 *
32 * FLEMatrixFast is a derived class that avoids a small overhead by not checking
33 * the index passed to operator [], operator (), and getColPtr for being too large.
34 * However, when debugging FLARRAY_ALWAYS_CHECK_INDICES can be defined to force index
35 checking with FLEMatrixFast too.
36 *
37 * FLENumMatrix is an extension of FLEMatrix which assumes elemType can do arithmetic. It supplies
38 * methods such as sum() which returns the summation of the elements.
39 *
40 * FLENumMatrixFast is exactly analogous to FLEMatrixFast. In fact I tried to have it multiple
41 * inherit from FLENumMatrix and FLEMatrixFast -- but I couldn't get that to work...
42 *
43 */

187 FLEMatrix<T> submatrix( const size_t startRow, const size_t endRow, const size_t startCol, const size_t endCol ) const{
188 // Return a copy of the submatrix from startRow to endRow and startCol to endCol (inclusive).
189 assert( startRow <= endRow );
190 assert( startCol <= endCol );
191 assert( validIndex( startRow, startCol ) );
192 assert( validIndex( endRow, endCol ) );
193 FLEMatrix<T> retVal( endRow - startRow + 1, endCol - startCol + 1 );
194 for( size_t i = startRow; i <= endRow; ++i ){
195 for( size_t j = startCol; j <= endCol; ++j ){
196 retVal( i-startRow, j-startCol ) = this->operator()( i, j );
197 }
198 }

327 enum dimensionT { ROW, COLUMN };
328 // dimension is really enum {0, 1}, choosing to sum over the current
329 // row or column.
330 T sum( dimensionT dimension, unsigned int index ){
331 assert( dimension < 2 );
332 assert( FLEMatrix<T>::m );
333 T ans = 0;
334 if( dimension==0 ){
335 unsigned int base = index;
336 for( size_t i = 0; i < FLEMatrix<T>::size0(); ++i ){
337 ans += FLEMatrix<T>::m[ base ];
338 base += FLEMatrix<T>::s1;
339 }
340 return ans;
341 }else{
342 unsigned int base = index * FLEMatrix<T>::s1;
343 for( size_t i = 0; i < FLEMatrix<T>::size1(); ++i ){
344 ans += FLEMatrix<T>::m[ base + i ];
345 }
346 }
347 return ans;
348 }
349
350 // dimension is really enum {0, 1}, choosing to divide from elements
351 // of the current row or column.
352 T divide( unsigned char dimension, unsigned int index, T divisor ){ // divide row or column.
353 assert( dimension < 2 );
354 T ans = 0;
355 if( dimension==0 ){
356 unsigned int base = index;
357 for( size_t i = 0; i < FLEMatrix<T>::size0(); ++i ){
358 FLEMatrix<T>::m[ base ] /= divisor;
359 base += FLEMatrix<T>::s1;
360 }
361 return ans;
362 }else{
363 unsigned int base = index * FLEMatrix<T>::s1;
364 for( size_t i = 0; i < FLEMatrix<T>::size1(); ++i ){
365 FLEMatrix<T>::m[ base + i ] /= divisor;
366 }
367 }
368 return ans;
369 }
370 void normalize( unsigned char dimension ){ // normalize row or column
371 assert( dimension < 2 );
372 size_t sumSize;
373 if( dimension == 0 ) sumSize = FLEMatrix<T>::size1();
374 else sumSize = FLEMatrix<T>::size0();
375 for( unsigned int i = 0; i < sumSize; ++i ){
376 divide( dimension, i, sum( dimension, i ) );
377 }
378 }
