27// if num >= max, output max.
28// else output the first number from the prime list not smaller than num.
29static int nextPrime( size_t num ){
30 const size_t max = 1796359771;
31 const size_t somePrimes[] = { 11, 19, 37, 73, 109, 163, 251, 367, 557, 823, 1237, 1861, 2777, 4177, 6247,
32 9371, 14057, 21089, 31627, 47431, 71143, 106721, 160073, 240101, 360163,
33 540217, 810343, 1215497, 1823231, 2734867, 4102283, 6153409, 9230113, 13845163,
34 20767751, 31151623, 46727449, 70091159, 105136753, 157705109, 236557751,
35 354836507, 532254761, 798382121, 1197573203, max };
36 if( num >= max ) return max;
37 size_t i;
38 for( i = 0; somePrimes[i] < num; ++i );
39 return somePrimes[i];
40}

42// designed so that permuting bytes gives different hash values.
43static size_t bytesHashVal( const size_t numBytes, size_t* a){
44 size_t numWords = numBytes/ sizeof(size_t);
45 size_t numExtraBytes = numBytes - sizeof(size_t)*numWords;
46 size_t key = 0;
47 size_t* pastLastWord = a+numWords;
48 for( ; a < pastLastWord; ++a ){
49 key ^= *a;
50 BIT_ROTATE_LEFT( key );
51 }
52 unsigned char* b = (unsigned char*) pastLastWord+1;
53 switch( numExtraBytes ){ // purposely using fallthrough...
54 case 3: key ^= ( *b++ << 16 );
55 case 2: key ^= ( *b++ << 8 );
56 case 1: key ^= *b;
57 default: return key;
58 }
59}

61// designed so that hash( s0, s1 ) != hash( s1, s0 ).
62static size_t string2HashVal( const std::string& s0, const std::string& s1 ){
63 size_t hash2 = bytesHashVal( s1.size(), (size_t*) s1.c_str() );
64 BIT_ROTATE_LEFT( hash2 );
65 return( bytesHashVal( s0.size(), (size_t*) s0.c_str() ) ^ hash2 );
66}

144public:

147 String2KeyHashIterator<ValT>( String2KeyHashtableLink<ValT>** head, String2KeyHashtableLink<ValT>** last ){
148 this->head = head; this->last = last;
149 if( head ) linkPtr = *head;
150 }
151 bool valid(){ return head; }
152 bool operator==( const String2KeyHashIterator<ValT>& ch ) const{
153 if( head != ch.head ) return false;
154 return( linkPtr == ch.linkPtr );
155 }
156 bool operator!=( const String2KeyHashIterator<ValT>& ch ) const{ return( !(*this==ch) ); }
157 String2KeyAndValue<ValT> * getElemPtr() const{ return getLinkPtr()->getElemPtr(); }
158 String2KeyAndValue<ValT>& operator*() const{ return *(getElemPtr()); }
159
160 String2KeyHashIterator<ValT>& operator++(){
161 if( !valid() ) return *this; // ++ing an invalid iterator is a no-op not an error.
162 if( linkPtr->getNext() ){ // walk down one step.
163 linkPtr = linkPtr->getNext();
164 return (*this);
165 }
166 if( head >= last ){
167 head = NULL; linkPtr = NULL;
168 return *this;
169 }
170 for( ++head; (head<last) && !*head; ++head );
171 if( *head ) linkPtr = *head;
172 else{ head = NULL; linkPtr = NULL; }
173
174 return *this;
175 }

178private:

180 String2KeyHashtableLink<ValT>** head;
181 String2KeyHashtableLink<ValT>* linkPtr;
182 String2KeyHashtableLink<ValT>** last;

294 ValT& add2( const std::string& s0, const std::string& s1, bool resizeAllowed = true ){
295#ifdef CBRC_DEBUG
296 assert( tableStateConsistent( tsize, t ) );
297#endif // debined CBRC_DEBUG
298 const unsigned int String2KeyHashtableMaxChainLen = 8;
299 size_t index = string2HashVal( s0, s1 ) % tsize;
300 unsigned int chainLen;
301 String2KeyHashtableLink<ValT>* linkPtr = t[index];
302 if( !linkPtr ){
303 // std::cout << "##add to head of list at position " << index << std::endl;
304 t[index] = new String2KeyHashtableLink<ValT>( s0, s1, defaultValue );
305 ++numElems;
306#ifdef CBRC_DEBUG
307 assert( tableStateConsistent( tsize, t ) );
308#endif // debined CBRC_DEBUG
309 return *(t[index]->getElemPtr());
310 }
311 // std::cout << "##head of list was not null so walk down\n";
312 for( chainLen = 2; !(linkPtr->elemPtr->keysEqual(s0, s1)) && linkPtr->getNext(); linkPtr=linkPtr->getNext(),
313 chainLen++ );
314 if( linkPtr->elemPtr->keysEqual( s0, s1 ) ){
315 // std::cout << "element already in table\n";
316 return *(linkPtr->getElemPtr());
317 }
318 // need to allocate new slot, but first check if we need to resize.
319 if( resizeAllowed && (tsize < maxTSize) && (chainLen > String2KeyHashtableMaxChainLen) ){
320#ifdef CBRC_DEBUG
321 assert( tableStateConsistent( tsize, t ) );
322#endif // defined CBRC_DEBUG
323 resize( tsize+1 );
324#ifdef CBRC_DEBUG
325 assert( tableStateConsistent( tsize, t ) );
326#endif // defined CBRC_DEBUG
327 return add2( s0, s1, false );
328 }
329 ++numElems;
330 linkPtr->setNext( new String2KeyHashtableLink<ValT>( s0, s1, defaultValue ) );
331#ifdef CBRC_DEBUG
332 assert( linkPtr->valid() );
333 assert( tableStateConsistent( tsize, t ) );
334#endif // defined CBRC_DEBUG
335 return *(linkPtr->next->getElemPtr());
336 }

380 if( tablePtr == t ){
381 if( tableSize != tsize ){
382 std::cout << "tableSize: " << tableSize << " not equal to tsize: " << tsize << "\n";
383 return false;
384 }
385 }
386 for( size_t i = 0; i < tsize; i++ ){
387 if( !t[i] ) continue; // NULL's okay
388 String2KeyHashtableLink<ValT>* linkPtr = t[i];
389 if( !linkPtr->valid() ){
390 return false;
391 }
392 for( ; linkPtr->getNext(); linkPtr = linkPtr->getNext() ){
393 if( !linkPtr->valid() ){
394 return false;
395 }
396 }
397 }
398 return true;

416 void resize( size_t size ){
417 size_t newSize = nextPrime( size );
418 // std::cout << "resizing, old_size: " << size << " new size: " << newSize << std::endl;
419 String2KeyHashtableLink<ValT>** newT = new String2KeyHashtableLink<ValT>*[newSize];
420 memset( newT, 0, sizeof(String2KeyHashtableLink<ValT>*)*newSize );

422 String2KeyHashIterator<ValT> nextIt = curIt;
