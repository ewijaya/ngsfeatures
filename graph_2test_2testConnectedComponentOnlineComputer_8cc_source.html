18void testConnectedComponentOnlineComputer( const bool& verify ){
19 MatrixGraph mg( std::cin );
20 std::cout << "num nodes is " << mg.numNodes() << std::endl;
21 ConnectedComponentOnlineComputer ccoc( mg.numNodes() );
22
23 if( verify ){
24 // Very primitive v^4 algorithm for transitive closure follows:

26 while( !done ){
27 done = true;
28 for( nodeIndexT k = 2; k < mg.numNodes(); ++k ){

30 if( !mg.hasEdge( j, k ) ) continue;
31 for( nodeIndexT i = 0; i < j; ++i ){
32 if( mg.hasEdge(i, j) ){
33 if( mg.hasEdge(j, k) ){
34 if( !mg.hasEdge(i, k) ){
35 done = false;
36 mg.setEdge( i, k );
37 }
38 }
39 }
40 }
