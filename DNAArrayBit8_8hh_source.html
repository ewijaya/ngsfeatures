27typedef unsigned char byte;
28static const byte setMask[] = {0x03, 0x0c, 0x30, 0xc0};
29static const byte clearMask[] = {0xfc, 0xf3, 0xcf, 0x3f};
30static const byte maxResidueIndex = 3;
31
32template<size_t len>
33class DNAArrayBit8{
34public:
35 DNAArrayBit8(){ DNAArrayBit8( 0 ); }
36 DNAArrayBit8( byte bases ) throw( std::length_error ) : bases(bases) {
37 if( len > 4 ) throw( std::length_error( "length of DNAArrayBit8 cannot exceed 4" ) );
38 }
39 DNAArrayBit8( ResidueIndexMap::arrayT residueIndices
40 /* binary values, \0, \1, not "a", "c"... */ ) throw( std::length_error ){
41 if( residueIndices.size() != length() ){
42 std::cerr << "residueIndices.size() = " << residueIndices.size() << " != ";
43 std::cerr << "length() = " << length();
44 throw std::length_error( "residueIndices.size() != length()" );
45 }
46 bases = 0;
47 for( size_t i = 0; i < length(); ++i ) set( i, residueIndices[i] );
48 }
49 byte const getBases(){ return bases; }
50 void setBases( byte bs ){ bases = bs; }
51
52 // automatic type conversion to size_t. Useful for use as array index.
53 operator size_t() const{ return static_cast<size_t>(bases); }
54
55 // == Does not throw exeception for non DNA string, say "zebra". Just returns false.
56 bool operator==( const char* const s ) const{
57 for( size_t i = 0; i < len; ++i ){
58 if( !DNAProperResidueIndexMap.validResidue(s[i]) ) return false;
59 if( (*this)(i) != DNAProperResidueIndexMap.toResidueIndex(s[i]) ) return false;
60 }
61 if( s[len] ) return false;
62 return true;
63 }
64
65 byte operator() ( const size_t index ) const throw(std::out_of_range){
66 if( index >= length() ){
67 std::cout << "index: " << index << " >= length(): " << length() << std::endl;
68 throw std::out_of_range( "index >= length()" );
69 }
70 byte retVal = bases & setMask[index];
71 retVal >>= index; retVal >>= index;
72 return retVal;
73 }
74 void set( const size_t index, const byte base ) throw(std::out_of_range,std::invalid_argument){
75 if( index >= length() ){
76 std::cout << "index: " << index << " >= length(): " << length() << std::endl;
77 throw std::out_of_range( "index >= length()" );
78 }
79 if( base > maxResidueIndex ){
80 std::cout << "base: " << static_cast<int>(base);
81 std::cout << " >= maxResidueIndex: " << static_cast<int>(maxResidueIndex) << std::endl;
82 throw std::invalid_argument( "base > maxResidueIndex" );
83 }
84 byte b = base;
85 b <<= index; b <<= index;
86 bases &= clearMask[index];
87 bases |= b;
88 }
89 DNAArrayBit8<len> substitute( const size_t index, const byte base ) const{
90 DNAArrayBit8<len> retVal(bases);
91 retVal.set( index, base );
92 return retVal;
93 }
94 std::string getIndices() const{
95 std::string retVal;
96 retVal.resize( len );
97 for( unsigned int i = 0; i < len; ++i ) retVal[i] = (*this)(i);
98 return retVal;
99 }
100 size_t length() const{ return len; }
101 void friend testDNAArrayBit8();
102private:
103 byte bases;
104};
