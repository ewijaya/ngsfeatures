22std::string size_tToBinaryRepresentationString( size_t a ){
23 const size_t numBitsPerByte = 8;

25

29 size_t curBit = wordBitSize - 1;
30
31 // fill in '1' bits, in order of increasingly significant bits
32 while( a > 0 ){
33

36 }

40 }
