33bool TransfacReader::readRecord(){
34 for( transfacLine = nextNonBlankLine();
35 true;
36 transfacLine = nextNonBlankLine() ){
37 BEGIN_PARSE_TRANFAC_LINE:
38 // std::cout << "in readRecord; " << transfacLine << std::endl;
39 if( transfacLine.empty() ) return false; // not more records.
40
41 if( labelInfo.isRecordSeparator( transfacLine ) ) return true;
42
43 if( labelInfo.isAccession( transfacLine ) ){
44 _accession = transfacLine.content();
45 continue;
46 }
47 if( labelInfo.isBindingFactor( transfacLine ) ){
48 _bindingFactor.set( transfacLine.content() );
49 continue;
50 }
51 if( labelInfo.isDescription( transfacLine ) ){
52 _description = transfacLine.content();
53 continue;
54 }
55 if( labelInfo.isName( transfacLine ) ){
56 _name = transfacLine.content();
57 continue;
58 }
59 if( labelInfo.isFreqMatrix( transfacLine ) ){
60 readFreqMatrix();
61 goto BEGIN_PARSE_TRANFAC_LINE; // readFreqMatrix reads one line past frequency matrix.
62 }
63 std::cerr << "** processing not yet coded for: " << transfacLine << std::endl;
64 }
65}

68// Sets frequency matrix.
69// Reads one line past frequence matrix in stream.
70// Upon return, class global transfacLine is set to that line.
71void TransfacReader::readFreqMatrix(){
72 _alphabet = perlish::join( perlish::split( transfacLine.content() ) );
73 _freqMatrix.clear();
74 for( transfacLine = nextNonBlankLine();
75 !transfacLine.empty();
76 transfacLine = nextNonBlankLine() ){
77 if( !labelInfo.isFreqMatrixBody( transfacLine ) ){
78 return;
79 }
80 std::vector<std::string> fields = perlish::split( transfacLine.content() );
81 if( fields.size() != 5 ){
82 std::cout << __FILE__ << ":" << __LINE__
83 << " Error: expected 5 fields in frequency matrix body tranfac line: "
84 << transfacLine << std::endl;
85 exit( -1 );
86 }
87 _consensusSequence += fields[4];
88 fields.pop_back();
89 _freqMatrix.push_back( perlish::toUnsignedInts( fields ) );
90 }
91 assert( 0 );
92}

95TransfacTextLine TransfacReader::nextNonBlankLine() const{
96 TransfacTextLine retval;

99 line.size();
