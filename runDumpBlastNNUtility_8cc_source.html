28 void dumpBlastNNUtility( std::istream& trainStream,
29 std::istream& utilityMatrixStream,
30 std::istream& blastResultsStream ){
31 Blast8ResultReader blastResults( blastResultsStream, true );
32 LabeledInstances lis( LabeledInstances::text, trainStream );
33 utilityMatrixT utilityMatrix( utilityMatrixStream );
34 BlastEValuesByClass blastClassifier( blastResults, lis );
35 std::vector< std::pair<size_t, double> > eValues;
36
37 for( size_t i = 0; i < lis.size(); ++i ){
38 if( !blastResults.has( lis(i).name() ) ) continue;
39 std::vector< std::pair<std::string, double> > hitList = blastClassifier.bestEValuePerClass( lis(i).name() );
40 eValues.push_back( std::make_pair<size_t, double>(i, hitList[0].second) );
41 }
42 std::stable_sort( eValues.begin(), eValues.end(),
43 stlUtils::pairSecondLess<size_t, double> );
44 double log2r = 1.0 / log( 2.0 );
45 // First do id's with blast hits.
46 for( size_t index = 0; index < eValues.size(); ++index ){
47 // classify by blast
48 size_t i = eValues[index].first;
49 std::vector< std::pair<std::string, double> > hitList = blastClassifier.bestEValuePerClass( lis(i).name() );
50
51 // take care of ties.
52 size_t numSameScore;
53 for( numSameScore = 1;
54 (numSameScore < hitList.size()) && hitList[numSameScore].second == hitList[0].second;
55 ++numSameScore );
56 if( hitList.size() > numSameScore ){ // check correctness.
57 assert( hitList[numSameScore-1].second == hitList[0].second );
58 assert( hitList[numSameScore].second != hitList[0].second );
59 }
60 std::string predictedClass = hitList[0].first;
61 if( numSameScore == 1 ){
62 predictedClass = hitList[0].first;
63 }else{
64 FLENumArray<size_t> classCounts( lis.numClasses(), (size_t)0 );
65 for( size_t j = 0; j < numSameScore; ++j ){
66 ++classCounts[ lis.classNameToIdx( hitList[j].first ) ];
67 }
68 size_t maxCount = classCounts.max();
69 // mask out non maximal counts
70 // using the fact that LabeledInstances promises
71 // to index class in order of most frequence first.
72 // would be better to consider the total utility instead of just using the majority class...
73 FLENumArray<size_t>::iterator bestPos = std::find( classCounts.begin(), classCounts.end(), maxCount );
74 predictedClass = lis.classIdxToName( bestPos - classCounts.begin() );
75 }
76 std::string className = lis.classIdxToName( lis(i).classIdx() );
77
78 std::cout << lis(i).name() << " " << className << " " << predictedClass << " " << utilityMatrix( className, predictedClass ) << " " << log2r * log( hitList[0].second );
79 std::cout << std::endl;
80 }
81
82 // now do id's without blast hits
83 for( size_t i = 0; i < lis.size(); ++i ){
84 if( blastResults.has( lis(i).name() ) ) continue;
85 std::string className = lis.classIdxToName( lis(i).classIdx() );
86
87 // print dummy value for blast
88 std::cout << lis(i).name() << " " << className << " " << "pass" << " " << "utilityUndefined eValueUndefined\n";
89 }
90
91 }
92}; // end namescape cbrc
