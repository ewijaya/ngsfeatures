8 * Description: Constructs html for aligned best hits to queries coming from pseudo-html input
9 * consisting of lines of text in which each line starts with a sequence id.
10 *
11 * Those ids are compared to the sequences in the querySequencesStream.
12 *
13 * Achtung! The order of id's in querySequencesStream and queryHtmlStream is assumed
14 * to be consistent.
15 *
16 * For each id in both query streams the sequences in dataSequencesStream are searched
17 * the best match. If a good best match is found the alignment is output to outputStream
18 * with an anchor like: <A name=id>id</A>.
19 *
20 * The queryHtmlStream is modified to contain a link to that anchor based on
21 * outputPathnameForLinking. The html new line "<BR>" is also added to lines of
22 * queryHtmlStream.
23 */

43 outputStream <<
44 htmlStrings::header( "alignment of query strings to close matches in WoLFPSORT dataset" );

46 std::cout << htmlStrings::header( "WoLFPSORT predictions" );

62 std::vector<LabeledSequence> querySequences;
63
64 seqSlurper.slurpLabeledSequences( querySequences,
65 querySequencesStream,
66 aminoResidueIndexMap );

69 // read in html lines. Do not skip lines that start with '#'
70 const std::vector<std::string> queryHtml
71 = perlish::slurpLines( queryHtmlStream, perlish::alwaysFalse );
72
73 // adjust loosenessFactor so that more careful (but time consuming) matching is done for
74 // small queries.
75 alignScoreT loosenessFactor = 512;
76 loosenessFactor -= querySequences.size() * 16;
77 if( loosenessFactor < 32 ) loosenessFactor = 32;
78
79 size_t querySequencesIndex = 0;
80 for( size_t i = 0; i < queryHtml.size(); ++i ){
81 std::vector<std::string> fields = perlish::split( "[ \t]+", queryHtml[i] );
82 std::string seqName = fields[0];
83 for( ; (querySequencesIndex < querySequences.size()) &&
84 (querySequences[ querySequencesIndex ].name() != seqName)
85 ; ++querySequencesIndex );
86 enum { DISSIMILAR, ALIGNABLE, IDENTICAL } matchDegree = DISSIMILAR;
87 SimilaritySearchableSequences::indexScorePairT bestMatchResult;
88 std::vector<size_t> identicalMatches;
89 identicalMatches.clear();
90 if( querySequencesIndex < querySequences.size() ){
91 identicalMatches = sss.identicalMatches( querySequences[i] );
92 if( identicalMatches.size() ) matchDegree = IDENTICAL;
93 else{
94 bestMatchResult = sss.bestMatch( querySequences[i], loosenessFactor );
95 if( bestMatchResult.first < sss.size() ){
96 matchDegree = ALIGNABLE;
97 }
98 }
99 }
100 if( matchDegree == IDENTICAL ){
101 std::cout << queryHtml[i]
102 << " <I>identical</I> to seq";
103 if( identicalMatches.size() > 2 ) std::cout << "s";
104 std::cout << ": ";
105 for( size_t matchIndex = 0; matchIndex < identicalMatches.size(); ++matchIndex ){
106 if( matchIndex ) std::cout << ", ";
107 std::cout << sss( identicalMatches[matchIndex] ).name() << ":"
108 << sss( identicalMatches[matchIndex] ).className();
109 }
110 std::cout << "<BR>" << std::endl;
111 }else if( matchDegree == ALIGNABLE ){
112 const LabeledSequence& matchingSequence = sss(bestMatchResult.first);
113 const std::string& matchingName = matchingSequence.name();
114 AminoScorePam120 ad;
115 SeqGlobalAffineAligner gas( ad,
116 querySequences[ querySequencesIndex ].residueIndices() );
117 Alignment bestAlignment = gas.alignment( sss(bestMatchResult.first).residueIndices() );
118 outputStream << htmlStrings::anchorName( seqName, seqName )
119 << " <I>vs</I> " << matchingName << " <BR>" << std::endl;
120 bestAlignment.outputHtmlMarkSame( outputStream, true /* mark different */ );
121 std::cout << queryHtml[i] << " "
122 << htmlStrings::anchorHrefBasic( outputPathnameForLinking, "similar", matchingName )
123 << " to seq " << matchingName
124 << " of class " << matchingSequence.className() << "<BR>" << std::endl;
125 }else{
126 std::cout << queryHtml[i] << "<BR>" << std::endl;
127 }
128 }
129 outputStream << htmlStrings::end();
130 std::cout << htmlStrings::end();
131 }
132}; // end namescape cbrc
