249double der_f_k(std::vector <double> &x_vect, std::vector <double> &m_vect, int kk ) {
250
251 double res = 0;
252
253
254 // precal
255 double tmp_sum = compute_sum_hx_vect( x_vect );
256
257
258 for (int jj = 0; jj < static_cast<int>(x_vect.size()); jj++) {
259 int tmp_bool = ( kk == jj );
260 double hx_j = h( x_vect[ jj ] );
261 double der_hx_k = der_h( x_vect[ kk ] );
262 double ckj = der_hx_k * ( tmp_sum * tmp_bool - hx_j ) / pow( tmp_sum, 2 );
263 res += m_vect[ jj ] * tmp_sum * ckj / hx_j;
264 }
265
266 return res;
267}

274
275double der_ent_k(std::vector <double> &x_vect, int kk ) {
276
277 double res = 0;
278
279 // precal
280 double tmp_sum = compute_sum_hx_vect( x_vect );
281
282
283 for (int jj = 0; jj < static_cast<int>(x_vect.size()); jj++) {
284 int tmp_bool = ( kk == jj );
285 double hx_j = h( x_vect[ jj ] );
286 double der_hx_k = der_h( x_vect[ kk ] );
287 double ckj = der_hx_k * ( tmp_sum * tmp_bool - hx_j ) / pow( tmp_sum, 2 );
288 if ( hx_j > 0 ) {
289 res += - ckj * ( 1 + log( hx_j / tmp_sum ) );
290 }
291 }
292
293 return res;
294}

337double backtrackingLineSearch( std::vector <double> &x_vect, std::vector <double> &m_vect, std::vector <double> &der_vect, double new_fx, double gamma ) {
338
339 //
340 double alpha = 0.1;
341 double beta = 0.5;
342
343 //
344 double t = 1;
345
346 //
347 vector <double> tmp_x_vect = alloc_vect( der_vect.size() );
348
349 double res_a;
350 double res_b;
351
352 //
353 while( 1 ) {
354
355 // x + t * ( - derX )
356 for (unsigned jj = 0; jj < tmp_x_vect.size(); jj++) {
357 tmp_x_vect[ jj ] = x_vect[ jj ] - t * der_vect[ jj ];
358 }
359
360 //
361 res_a = final_f( tmp_x_vect, m_vect, gamma );
362 res_b = new_fx + alpha * t * ( - compute_dot( der_vect, der_vect ) );
363
364 // sortie normale
365 if ( res_a <= res_b ) {
366 break;
367 }
368
369 // maj
370 t *= beta;
371
372 // sortie anormale
373 if ( t == 0 ) {
374 break;
375 }
376
377 }
378
379
380 return t;
381
382}

388vector <double> gradient_descent( std::vector <double> &m_vect, double gamma ) {
389
390 //
391 //vector <double> x_vect = alloc_vect( m_vect.size() );
392 vector <double> new_x_vect = alloc_vect( m_vect.size() );
393 vector <double> der_vect = alloc_vect( m_vect.size() );
394
395
396 // Generate Random Number for x_vect
397 vector <double> x_vect;
398 for (int i = 0; i < static_cast<int>(m_vect.size()); i++) {
399
400 x_vect.push_back(sin(rand())*5);
401
402 }
403
404 //
405 int n_iter = 0;
406 double new_fx = final_f( x_vect, m_vect, gamma );
407 double old_fx = 0;
408 double epsilon = 0;
409
410 // param
411 int n_iter_max = 1000;
412 double nu = pow( 10, -4 );
413
414
415 //
416 while( 1 ) {
417
418 // gradient
419 final_der( x_vect, m_vect, gamma, der_vect );
420
421 // epsilon
422 epsilon = backtrackingLineSearch( x_vect, m_vect, der_vect, new_fx, gamma );
423
424 // maj x
425 for (unsigned jj = 0; jj < x_vect.size(); jj++) {
426 new_x_vect[ jj ] = x_vect[ jj ] - epsilon * der_vect[ jj ];
427 }
428 // maj fx
429 new_fx = final_f( new_x_vect, m_vect, gamma );
430
431 // sortie anormale
432 if ( n_iter > 0 && new_fx > old_fx ) {
433 break;
434 }
435
436 // sortie normale
437 if ( ( n_iter > 0 && abs( ( new_fx - old_fx ) / old_fx ) < nu * x_vect.size() ) || ( n_iter == n_iter_max ) ) {
438 break;
439 }
440
441 // maj
442 copy_vect( new_x_vect, x_vect );
443 n_iter += 1;
444 old_fx = new_fx;
445
446 }
447
448 //
449 //cout << "gradient_descent() : " << n_iter << " iterations" << endl;
450
451 //
452 return new_x_vect;
453}
