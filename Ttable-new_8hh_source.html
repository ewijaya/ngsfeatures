39long double betai(const long double& a, const long double& b, const long double& x) {
40
41 const int MAXK = 200;
42 int k;
43 long double p1, q1, p2, q2, d, previous;
44
45 if(a <= 0L) throw std::string("bad a in betai.");
46 if(b <= 0L) {
47 if(x < 1L) return 0L;
48 if(x == 1L) return 1L;
49 throw std::string("bad x in betai.");
50 }
51 if(x > (a + 1L) / (a + b + 2L))
52 return 1L - betai(b, a, 1L - x);
53 if(x <= 0L) return 0L;
54 p1 = 0L; q1 = 1L;
55 p2 = expl(a * logl(x) + b * logl(1L - x) + lgammal(a + b) - lgammal(a) - lgammal(b)) / a;
56 q2 = 1L;
57 for(k = 0; k < MAXK; ) {
58 previous = p2;
59 d = - (a + (_LD)k) * (a + b + (_LD)k) * x / ((a + 2L * (_LD)k) * (a + 2L * (_LD)k + 1L));
60 p1 = p1 * d + p2; q1 = q1 * d + q2;
61 k++;
62 d = (_LD)k * (b - (_LD)k) * x / ((a + 2L * (_LD)k - 1L) * (a + 2L * (_LD)k));
63 p2 = p2 * d + p1; q2 = q2 * d + q1;
64 if(q2 == 0L) {
65 p2 = std::numeric_limits<long double>::max(); continue;
66 }
67 p1 /= q2; q1 /= q2; p2 /= q2; q2 = 1L;
68 if(p2 == previous) return p2;
69 }
70 std::cerr << "a:" << a << " b:" << b << "x:" << x << std::endl;
71 throw std::string("Not converged.");
72 return p2;
73
74}
