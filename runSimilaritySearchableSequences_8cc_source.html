60 if( result.first != sss.size() ){

62 Alignment bestAlignment = gas.alignment( sss(result.first).residueIndices() );
63
64 std::cout << "best hit: " << sss.seqName( result )
65 << " score: " << result.second << std::endl;

67 bestAlignment.outputMarkSame( outputFormat, std::cout, markDifferent );
