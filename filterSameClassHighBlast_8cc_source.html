23 void filterSameClassHighBlast( std::istream& trainStream,
24 std::istream& blastResultsStream,
25 const double& eValueThreshold ){
26 Blast8ResultReader blastResults( blastResultsStream, false );
27 LabeledInstances lis( LabeledInstances::text, trainStream );
28 std::map<std::string, std::string> nameToClass = lis.getNameToClassMap();
29 std::vector<std::string> hitNames;
30
31 for( size_t i = 0; i < lis.size(); ++i ){
32 if( !blastResults.has( lis(i).name() ) ) continue;
33
34 const std::vector<Blast8Result> hitV = blastResults( lis(i).name() );
35
36 hitNames.clear();
37
38 for( size_t j = 0; j < hitV.size(); ++j ){
39
40 if(( nameToClass[ hitV[j].hitName() ] == nameToClass[ lis(i).name() ] )
41 && ( hitV[j].eValue() <= eValueThreshold )
42 ){
43 if( hitV[j].hitName() != lis(i).name() ) hitNames.push_back( hitV[j].hitName() );
44 }
45 }
46
47 if( hitNames.size() > 0 ){
48 std::cout << lis(i).name();
49 for( size_t i = 0; i < hitNames.size(); ++i ) std::cout << " " << hitNames[i];
50 std::cout << std::endl;
51 }
52 }
53 }
54}; // end namescape cbrc
