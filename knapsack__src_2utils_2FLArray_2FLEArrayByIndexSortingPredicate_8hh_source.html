29inline FLEArray<size_t> FLEArrayIdentityPermutation( size_t s ){
30 FLEArray<size_t> ip( s );
31 for( size_t i = 0; i < s; ++i ) ip[i] = i;
32 return ip;
33}
34
35template < typename T, typename op=std::less<T> >
36class FLEArrayByIndexSortingPredicate{
37public:
38 FLEArrayByIndexSortingPredicate( const FLEArray<T>& a ) : a(a){}
39 bool operator()( const size_t index0, const size_t index1 ) const{
40 return op()( a[index0], a[index1] );
41 }
42private:
43 const FLEArray<T>& a;
44};

46template <typename T>
47FLEArray<size_t> FLEArraySortedPermutation( FLEArray<T> a ){
48 FLEArray<size_t> perm = FLEArrayIdentityPermutation( a.size() );
49 std::stable_sort( perm.begin(), perm.end(), FLEArrayByIndexSortingPredicate<T>( a ) );
50 return perm;
51}

53template <typename T, typename op>
54FLEArray<size_t> FLEArraySortedPermutation( FLEArray<T> a ){
55 FLEArray<size_t> perm = FLEArrayIdentityPermutation( a.size() );
56 std::stable_sort( perm.begin(), perm.end(), FLEArrayByIndexSortingPredicate<T,op>( a ) );
57 return perm;
58}
