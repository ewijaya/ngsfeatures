24 // Pick random element by uniform distribution
25 // If called with non-zero callerGivenSeed reseed with that value
26 // Otherwise seed only on first call
27 template<typename T>
28 const T& randomElement( const FLEArray<T> a, long int callerGivenSeed = 0 ){
29 static long int seed = 0;
30
31 /* ***** Seed if necessary ***** */
32 if( callerGivenSeed ){
33 seed = callerGivenSeed;
34 srand48( seed );
35 }else if( !seed ){
36 seed = getpid() * (long) time( NULL );
37 srand48( seed );
38 }
39
40 size_t randomIndex = static_cast<long int>( drand48() * a.size() );
41 return( a[ randomIndex ] );
42 };

47 // upon return numCombinationsVector[i] holds n choose i, where n = numCombinationsVector.size() - 1.
48 template<typename elemType>
49 void setNumCombinationsVector( FLENumArray<elemType>& numCombinationsVector ){
50
51 numCombinationsVector[0] = 1;
52 unsigned int n = numCombinationsVector.size() - 1;
53 for( unsigned int i = 1; i <= n; ++i ){
54 numCombinationsVector[i] = numCombinationsVector[i-1] * (n-i+1) / i;
55 }
56 }

79 // set a to cumulative sum of b.
80 template <typename T>
81 void cumulativeSum( FLENumArray<T>& a, const FLENumArray<T>& b ){
82 if( !b.size() ) return;
83 if( !a.size() ) a.setSize( b.size() );
84 GDB_ASSERT( a.size() == b.size() );
85 a[0] = b[0];
86 for( size_t i = 1; i < b.size(); ++i ) a[i] = a[i-1] + b[i];
87 }
88
89 // normalize elements of a to add to sum.
90 template <typename T>
91 void normalizeSum( FLENumArray<T>& a, double sumAfterNormalization = 1.0 ){
92 double sum = (double) a.sum();
93 double ratio = sumAfterNormalization / sum;
94 for( size_t i = 0; i < a.size(); ++i ) a[i] = a[i] * ratio;
95 }
