14void ConstantLabeledSquareDoubleFLENumMatrix::setFromStream( std::istream& is,
15 const std::string inputDelimiter ){
16 std::string format( "first line is: label0 label1 ... labelN\n"
17 "next line contains the vector for label0 separated by spaces"
18 "following line contains the vector for label1 separated by spaces"
19 "...\n"
20 "empty lines and lines starting with \"#\" are ignored" );
21 std::string line = perlish::slurpLine( is, perlish::isSharpCommentOrEmpty );
22 std::vector< std::string > fields = perlish::split( inputDelimiter, line );
23
24 if( fields.size() < 2 ){
25 std::cout << __FILE__ << ":" << __LINE__ << " Error: ";
26 std::cout << " not enough fields in first line. Format is:\n" << format << std::endl;
27 assert( 0 );
28 }
29
30 for( size_t i = 0; i < fields.size(); ++i ){
31 GDB_ASSERTF( !labels().has( fields[i] ),
32 "the field name \"%s\" appears to be duplicated", fields[i].c_str() );
33 _labels.getSerialNumber( fields[i] );
34 }
35
36 m.setSize( labels().size(), labels().size() );
37
38 for( size_t i = 0; i < labels().size(); ++i ){
39 line = perlish::slurpLine( is, perlish::isSharpCommentOrEmpty );
40 fields = perlish::split( inputDelimiter, line );
41 if( fields.size() != labels().size() ){
42 std::cout << " not enough fields in line: " << line << "\nFormat is:\n" << format << std::endl;
43 assert( 0 );
44 }
45 for( size_t j = 0; j < fields.size(); ++j ) m( i, j ) = strtod( fields[j].c_str(), NULL );
46 }
47}

50/* labelsNewOrder should be a permutation of a subset of the labels.
51 * the rows and column are reordered such that the labels contained in labelsNewOrder come first.
52 * if there are labels not contained in labelsNewOrder they will remain accessible, but their
53 * indices will start with labelsNewOrder.size()
54 */
55void ConstantLabeledSquareDoubleFLENumMatrix
56::renumLabels( const SerialNumbersForStrings<size_t>& labelsNewOrder ){
57
58 /* ***** Construct new SerialNumbersForStrings for labels ***** */
59 SerialNumbersForStrings<size_t> newLabels( labelsNewOrder );
60
61 // add labels in this->labels, to end if not already present in labelsNewOrder
62 for( size_t i = 0; i < labels().size(); ++i ){
63 newLabels.getSerialNumber( labels()(i) );
64 }
65
66 // die if labelsNewOrder was not a subset of this->labels
67 if( labels().size() != newLabels.size() ){
68 _renumLabels_die( newLabels );
69 }
70
71
72 /* ***** transfer elements to the a new matrix with the new ordering of labels ***** */
73 FLENumMatrixFast<double> newM( m.size0(), m.size1() );
74
75 for( size_t i = 0; i < newLabels.size(); ++i ){
76 for( size_t j = 0; j < newLabels.size(); ++j ){
77 newM( i, j ) = (*this)( newLabels(i), newLabels(j) );
78 }
79 }
80
81 // copy new matrix to this object data
82 m = newM;
83 _labels = newLabels;
84} // end renumLabels
