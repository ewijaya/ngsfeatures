18void MatrixGraph::readContinuousIntegerNodesFormat( std::istream& is ){
19
20 std::string line;
21 std::vector<std::string> fields;
22 nodeIndexT startIndex = 0; // init to silence compiler warning.
23
24
25 /* ***** read num nodes declaration line ***** */
26 std::string numNodeDeclarationLine = perlish::slurpLine( is );
27 assert( numNodeDeclarationLine.size() );
28
29 fields = perlish::split( inputDelimiter(), numNodeDeclarationLine );
30
31 assert( fields.size() == 2 );
32 startIndex = atoi( fields[0].c_str() );
33 _numNodes = atoi( fields[1].c_str() ) - startIndex + 1;
34
35 // add node names as strings

37 std::string nodeName( boost::lexical_cast<std::string>( startIndex + i ) );
38 labels.registerNewSerialNumber( nodeName );

40
41 edges.setSize( _numNodes );
42 edges.zero();
43
44
45 /* ***** read in edges ***** */
46 while( !is.eof() ){
47 std::string curLine = perlish::slurpLine( is );
48
49 fields = perlish::split( inputDelimiter(), curLine );
50
51 if( fields.size() < 2 ){
52 std::cerr << "skipping line: " << curLine << std::endl;
53 continue;
54 }
55
56 nodeIndexT n0 = atoi( fields[0].c_str() ) - startIndex;
57 for( size_t i = 1; i < fields.size(); ++i ){
58 nodeIndexT n1 = atoi( fields[i].c_str() ) - startIndex;
59 setEdge( n0, n1 );
60 }

66void MatrixGraph::readEdgesOneEdgePerLineFormat( std::istream& is ){
67 while( !is.eof() ){
68 std::string line = perlish::slurpLine( is );
69 if( line.size() == 0 ) break;
70
71 std::vector<std::string> fields = perlish::split( inputDelimiter(), line );
72
73 if( !fields.size() == 2 ){
74 std::cout << "Format error. Expected two fields when reading line: " << line << std::endl;
75 exit( -1 );
76 }
77 if( fields[0] == fields[1] ){
78 std::cerr << "Warning. Skipping self edge from line: " << line << std::endl;
79 continue;
80 }
81 nodeIndexT n0 = labels.getSerialNumber( fields[0] );
82 nodeIndexT n1 = labels.getSerialNumber( fields[1] );
83 setEdge( n0, n1 );
84 }
85}
